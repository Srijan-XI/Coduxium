📚 Introduction

What is SQLite?

SQLite is a C-library that provides a lightweight, disk-based database that doesn't require a separate server process. It allows accessing the database using a nonstandard variant of SQL. SQLite is completely self-contained, requiring no external dependencies, and stores the entire database in a single cross-platform file.

Key Features

📦 Self-Contained — No external dependencies
🚫 Serverless — No server process needed
⚙️ Zero-Configuration — No setup required
📁 Single File — Entire DB in one file
🌐 Cross-Platform — Works everywhere
⚡ Fast — Lightweight and quick
🔓 Public Domain — No licensing concerns
🔒 ACID Compliant — Reliable transactions

Why Choose SQLite?

  • Simplicity: No installation, no configuration
  • Portability: Single file, easy to share/backup
  • Embedded: Perfect for mobile apps, browsers
  • Testing: Great for development and testing
  • Small Apps: Ideal for low to medium traffic
  • Offline Apps: Works without network connection

Common Use Cases

⚠️ Limitations: Not suitable for high-write concurrency or network file systems.

⚙️ Installation & Setup

SQLite comes pre-installed on most systems, but you can also install it manually.

Windows Installation

1
Download SQLite Tools

Visit sqlite.org/download.html and download the precompiled binaries for Windows.

2
Extract Files

Extract the ZIP file to a folder (e.g., C:\sqlite).

3
Add to PATH

Add the folder to your system PATH environment variable.

4
Verify Installation 
sqlite3 --version

macOS Installation

SQLite comes pre-installed on macOS. To update:

# Using Homebrew
brew install sqlite3

# Verify
sqlite3 --version

Linux Installation

Ubuntu/Debian

sudo apt update
sudo apt install sqlite3

sqlite3 --version

Fedora/RHEL

sudo dnf install sqlite

sqlite3 --version

Arch Linux

sudo pacman -S sqlite

sqlite3 --version

Create First Database

# Open SQLite CLI and create database
sqlite3 mydata base.db

# SQLite prompt
sqlite> .databases
sqlite> .tables
sqlite> .quit

📖 User Guide

SQLite CLI Commands

-- Dot commands (SQLite specific)
.help              -- Show help
.databases         -- List attached databases
.tables            -- List tables
.schema tablename  -- Show table schema
.mode column       -- Set output mode
.headers on        -- Show column headers
.quit              -- Exit SQLite

Create Tables

-- Create table
CREATE TABLE users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    username TEXT NOT NULL UNIQUE,
    email TEXT NOT NULL UNIQUE,
    password_hash TEXT NOT NULL,
    created_at TEXT DEFAULT CURRENT_TIMESTAMP
);

-- Create table with foreign key
CREATE TABLE posts (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    title TEXT NOT NULL,
    content TEXT,
    created_at TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- Show table structure
.schema users

CRUD Operations

-- INSERT
INSERT INTO users (username, email, password_hash)
VALUES ('john_doe', 'john@example.com', 'hashed_password');

-- Multiple inserts
INSERT INTO users (username, email, password_hash) VALUES
    ('alice', 'alice@example.com', 'hash1'),
    ('bob', 'bob@example.com', 'hash2');

-- SELECT
SELECT * FROM users;
SELECT username, email FROM users WHERE id = 1;
SELECT * FROM users WHERE username LIKE 'john%';

-- UPDATE
UPDATE users SET email = 'newemail@example.com' WHERE id = 1;

-- DELETE
DELETE FROM users WHERE id = 5;

Transactions

-- Begin transaction
BEGIN TRANSACTION;

INSERT INTO users (username, email, password_hash)
VALUES ('test', 'test@example.com', 'hash');

UPDATE users SET email = 'updated@example.com' WHERE username = 'test';

-- Commit or rollback
COMMIT;
-- or
ROLLBACK;

Joins & Aggregation

-- INNER JOIN
SELECT u.username, p.title
FROM users u
INNER JOIN posts p ON u.id = p.user_id;

-- LEFT JOIN
SELECT u.username, COUNT(p.id) as post_count
FROM users u
LEFT JOIN posts p ON u.id = p.user_id
GROUP BY u.id;

-- Aggregation functions
SELECT COUNT(*) FROM users;
SELECT AVG(id) FROM users;
SELECT MAX(created_at), MIN(created_at) FROM users;

Indexes

-- Create index
CREATE INDEX idx_username ON users(username);
CREATE INDEX idx_email ON users(email);

-- Create unique index
CREATE UNIQUE INDEX idx_unique_email ON users(email);

-- List indexes
.indexes users

-- Drop index
DROP INDEX idx_username;

🚀 Advanced Topics

Full-Text Search

-- Create FTS table
CREATE VIRTUAL TABLE articles_fts USING fts5(title, content);

-- Insert data
INSERT INTO articles_fts (title, content) VALUES
    ('Introduction to SQLite', 'SQLite is a lightweight database...'),
    ('Advanced SQL', 'Learn advanced SQL techniques...');

-- Search
SELECT * FROM articles_fts WHERE articles_fts MATCH 'SQLite';
SELECT * FROM articles_fts WHERE content MATCH 'database';

JSON Support (SQLite 3.38+)

-- Create table with JSON
CREATE TABLE products (
    id INTEGER PRIMARY KEY,
    name TEXT,
    attributes TEXT -- JSON data
);

-- Insert JSON
INSERT INTO products (name, attributes) VALUES
    ('Laptop', json('{"color": "silver", "weight": 1.5}'));

-- Extract JSON values
SELECT 
    name,
    json_extract(attributes, '$.color') as color,
    json_extract(attributes, '$.weight') as weight
FROM products;

-- Validate JSON
SELECT json_valid('{"key": "value"}');  -- Returns 1 (true)

Backup & Restore

# Backup database
sqlite3 mydatabase.db ".backup backup.db"

# Dump to SQL file
sqlite3 mydatabase.db ".dump" > backup.sql

# Restore from SQL file
sqlite3 newdatabase.db < backup.sql

# Vacuum (shrink database)
sqlite3 mydatabase.db "VACUUM;"

Application Integration

// Node.js (better-sqlite3)
const Database = require('better-sqlite3');
const db = new Database('mydata.db');

// Create table
db.exec(`CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL
)`);

// Insert
const insert = db.prepare('INSERT INTO users (name) VALUES (?)');
insert.run('John Doe');

// Query
const users = db.prepare('SELECT * FROM users').all();
console.log(users);

// Python (sqlite3)
import sqlite3

conn = sqlite3.connect('mydata.db')
cursor = conn.cursor()

cursor.execute('''CREATE TABLE IF NOT EXISTS users
                  (id INTEGER PRIMARY KEY, name TEXT)''')

cursor.execute('INSERT INTO users (name) VALUES (?)', ('Alice',))
conn.commit()

cursor.execute('SELECT * FROM users')
rows = cursor.fetchall()

conn.close()

// Rust (rusqlite)
use rusqlite::{Connection, Result};

fn main() -> Result<()> {
    let conn = Connection::open("mydata.db")?;
    
    conn.execute(
        "CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT)",
        [],
    )?;
    
    conn.execute("INSERT INTO users (name) VALUES (?1)", ["Alice"])?;
    
    Ok(())
}

Performance Optimization

-- Enable foreign keys
PRAGMA foreign_keys = ON;

EXPLAIN QUERY PLAN SELECT * FROM users WHERE email = 'test@example.com';

-- Optimize database
VACUUM;
ANALYZE;
