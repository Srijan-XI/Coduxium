📚 Introduction

What is MariaDB?

MariaDB is a community-developed, commercially supported fork of MySQL created by the original developers of MySQL in 2009. It's designed to maintain compatibility with MySQL while offering additional features, better performance, and ensuring the database remains free and open-source under the GPL.

Key Features

🔄 MySQL Compatible — Drop-in replacement
⚡ Performance — Faster query execution
🔓 Open Source — Truly free, GPL licensed
🗄️ Storage Engines — Multiple engine support
🔐 Security — Enhanced authentication
🔄 Replication — Master-slave, Galera cluster
📊 Analytics — ColumnStore for analytics
🌐 JSON Support — Native JSON functions

Why Choose MariaDB?

  • MySQL Compatibility: Easy migration from MySQL
  • Community-Driven: No Oracle ownership concerns
  • Better Performance: Optimizer improvements, faster queries
  • More Features: Window functions, CTEs, JSON improvements
  • Enterprise Ready: Used by major corporations
  • Active Development: Regular releases with new features
⚠️ Note: While highly compatible, MariaDB 10.6+ has diverged from MySQL in some features.

📖 User Guide

Database Operations

-- Create database
CREATE DATABASE ecommerce;

-- Use database
USE ecommerce;

-- Show databases
SHOW DATABASES;

-- Drop database
DROP DATABASE IF EXISTS old_db;

Table Operations

-- Create table
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_email (email)
);

-- Show tables
SHOW TABLES;

-- Describe table
DESCRIBE users;

-- Alter table
ALTER TABLE users ADD COLUMN last_login TIMESTAMP;

-- Drop table
DROP TABLE IF EXISTS old_table;

CRUD Operations

-- INSERT
INSERT INTO users (username, email, password_hash)
VALUES ('john_doe', 'john@example.com', 'hashed_password');

-- Multiple inserts
INSERT INTO users (username, email, password_hash) VALUES
    ('alice', 'alice@example.com', 'hash1'),
    ('bob', 'bob@example.com', 'hash2');

-- SELECT
SELECT * FROM users;
SELECT username, email FROM users WHERE id = 1;
SELECT * FROM users WHERE created_at > '2024-01-01';

-- UPDATE
UPDATE users SET email = 'newemail@example.com' WHERE id = 1;

-- DELETE
DELETE FROM users WHERE id = 5;

-- Transactions
START TRANSACTION;
UPDATE users SET email = 'test@example.com' WHERE id = 1;
COMMIT;
-- or
ROLLBACK;

Joins

-- Create orders table
CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    total DECIMAL(10,2),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

-- INNER JOIN
SELECT u.username, o.total, o.created_at
FROM users u
INNER JOIN orders o ON u.id = o.user_id;

-- LEFT JOIN
SELECT u.username, COUNT(o.id) as order_count
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
GROUP BY u.id;

Advanced Queries

-- Window Functions (MariaDB 10.2+)
SELECT 
    username,
    total,
    ROW_NUMBER() OVER (ORDER BY total DESC) as rank
FROM users u
JOIN orders o ON u.id = o.user_id;

-- Common Table Expressions (CTE)
WITH high_value_orders AS (
    SELECT user_id, SUM(total) as total_spent
    FROM orders
    GROUP BY user_id
    HAVING total_spent > 1000
)
SELECT u.username, h.total_spent
FROM users u
JOIN high_value_orders h ON u.id = h.user_id;

-- JSON Functions
CREATE TABLE products (
    id INT PRIMARY KEY,
    name VARCHAR(100),
    attributes JSON
);

INSERT INTO products VALUES
(1, 'Laptop', '{"color": "silver", "weight": 1.5}');

SELECT 
    name,
    JSON_VALUE(attributes, '$.color') as color
FROM products;

🚀 Advanced Topics

Indexes & Performance

-- Create index
CREATE INDEX idx_username ON users(username);

CREATE FULLTEXT INDEX idx_description ON products(description);

Replication Setup

-- Master configuration (my.cnf)
[mysqld]
server-id=1
log-bin=mysql-bin
binlog-format=ROW

-- Restart MariaDB
sudo systemctl restart mariadb

-- Create replication user
CREATE USER 'repl'@'%' IDENTIFIED BY 'password';
GRANT REPLICATION SLAVE ON *.* TO 'repl'@'%';

-- Get master status
SHOW MASTER STATUS;

-- Slave configuration (my.cnf)
[mysqld]
server-id=2

-- Configure slave
CHANGE MASTER TO
  MASTER_HOST='master_ip',
  MASTER_USER='repl',
  MASTER_PASSWORD='password',
  MASTER_LOG_FILE='mysql-bin.000001',
  MASTER_LOG_POS=154;

-- Start slave
START SLAVE;

-- Check slave status
SHOW SLAVE STATUS\G

Backup & Restore

# Backup single database
mysqldump -u root -p ecommerce > ecommerce_backup.sql

# Backup all databases
mysqldump -u root -p --all-databases > all_backup.sql

# Backup with compression
mysqldump -u root -p ecommerce | gzip > ecommerce_backup.sql.gz

# Restore database
mysql -u root -p ecommerce < ecommerce_backup.sql

# Restore compressed backup
gunzip < ecommerce_backup.sql.gz | mysql -u root -p ecommerce

User Management

-- Create user
CREATE USER 'appuser'@'localhost' IDENTIFIED BY 'secure_password';

-- Grant privileges
GRANT SELECT, INSERT, UPDATE, DELETE ON ecommerce.* TO 'appuser'@'localhost';

-- Grant all privileges
GRANT ALL PRIVILEGES ON ecommerce.* TO 'admin'@'localhost';

-- Show grants
SHOW GRANTS FOR 'appuser'@'localhost';

-- Revoke privileges
REVOKE INSERT ON ecommerce.* FROM 'appuser'@'localhost';

-- Drop user
DROP USER 'olduser'@'localhost';

-- Change password
ALTER USER 'appuser'@'localhost' IDENTIFIED BY 'new_password';

Application Integration

const [rows] = await pool.query('SELECT * FROM users WHERE id = ?', [1]);

import mysql.connector

conn = mysql.connector.connect(
    host='localhost',
    user='root',
    password='password',
    database='ecommerce'
)

cursor = conn.cursor()
cursor.execute('SELECT * FROM users')
results = cursor.fetchall()
