📚 Introduction

What is MySQL?

MySQL is a powerful, open-source relational database management system (RDBMS) that has been the backbone of countless web applications for over 25 years. Known for its reliability, ease of use, and proven performance, MySQL powers everything from small personal projects to large-scale enterprise applications.

Key Features

🔒 ACID Compliance — Guaranteed data integrity with transactions
⚡ High Performance — Optimized for fast read and write operations
🔄 Replication — Master-slave and group replication support
💾 InnoDB Engine — MVCC, row-level locking, foreign keys
📊 Stored Procedures — Business logic in the database
🔍 Full-Text Search — Built-in text search capabilities
🌐 Cross-Platform — Runs on Windows, Linux, macOS
🛡️ Security — SSL support, encryption, access control

Why Choose MySQL?

  • Proven Reliability: Used by industry giants like Facebook, Twitter, YouTube, and Netflix for mission-critical applications.
  • Easy to Learn: Standard SQL syntax makes it accessible for beginners while offering advanced features for experts.
  • Scalability: Handles everything from small databases to petabyte-scale data warehouses.
  • Performance: Optimized query execution, efficient indexing, and query caching deliver fast response times.
  • Community & Support: Large community, extensive documentation, and commercial support available through Oracle.
  • Cost-Effective: Open-source with a flexible licensing model (GPL and commercial).

Use Cases

🌐 Web Applications — CMS, forums, e-commerce platforms
💼 Enterprise Software — ERP, CRM, business applications
📊 Data Warehousing — Analytics, reporting, BI tools
📱 Mobile Apps — Backend data storage and APIs
🎮 Gaming — User data, leaderboards, game state
📰 Content Management — WordPress, Drupal, Joomla

InnoDB Storage Engine

InnoDB is the default storage engine in MySQL, providing:

  • ACID Transactions: Full support for commit, rollback, and crash recovery
  • Row-Level Locking: Better concurrency than table-level locks
  • Foreign Keys: Maintain referential integrity across tables
  • MVCC: Multi-Version Concurrency Control for consistent reads
  • Crash Recovery: Automatic recovery after unexpected shutdowns
⚠️ Important: Always use the InnoDB storage engine for production applications. MyISAM lacks transaction support and crash recovery capabilities.

⚙️ Installation & Setup

Get started with MySQL by installing MySQL Server and client tools on your platform.

Windows Installation

Option 1 — MySQL Installer (Recommended)

1 Download MySQL Installer from MySQL Downloads
2 Run the installer and choose "Developer Default" or "Server only"
3 Select products: MySQL Server, MySQL Shell, MySQL Workbench
4 Configure server: Set root password, create user accounts
5 Choose to run MySQL as a Windows Service

Option 2 — winget

# Install MySQL Server
winget install Oracle.MySQL

# Verify installation
mysql --version

# Access MySQL (will prompt for password)
mysql -u root -p

Option 3 — Chocolatey

# Install MySQL
choco install mysql

# Start MySQL service
net start MySQL80

# Connect to MySQL
mysql -u root -p

macOS Installation

# Install MySQL using Homebrew
brew install mysql

# Start MySQL service
brew services start mysql

# Secure the installation (set root password, remove test databases)
mysql_secure_installation

# Connect to MySQL
mysql -u root -p

# Stop MySQL service
brew services stop mysql

Linux Installation

Ubuntu/Debian

# Update package index
sudo apt update

# Install MySQL Server
sudo apt install mysql-server

# Start MySQL service
sudo systemctl start mysql
sudo systemctl enable mysql

# Secure installation
sudo mysql_secure_installation

sudo systemctl status mysql

# Install MySQL Server

# Get temporary root password
sudo grep 'temporary password' /var/log/mysqld.log

# Secure installation (will ask to change root password)
sudo mysql_secure_installation

# Connect to MySQL
mysql -u root -p

Database Management

USE myapp;

DROP DATABASE IF EXISTS old_database;

Create Tables

-- Create users table
CREATE TABLE users (
  id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  email VARCHAR(255) NOT NULL UNIQUE,
  username VARCHAR(50) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  first_name VARCHAR(100),
  last_name VARCHAR(100),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  is_active BOOLEAN DEFAULT TRUE,
  
  INDEX idx_email (email),
  INDEX idx_username (username),
  INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Create orders table with foreign key
CREATE TABLE orders (
  id BIGINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
  user_id BIGINT UNSIGNED NOT NULL,
  order_number VARCHAR(50) NOT NULL UNIQUE,
  total DECIMAL(10, 2) NOT NULL,
  status ENUM('pending', 'processing', 'completed', 'cancelled') DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  INDEX idx_user_id (user_id),
  INDEX idx_status (status),
  INDEX idx_created_at (created_at)
) ENGINE=InnoDB;

-- Show table structure
DESCRIBE users;
SHOW CREATE TABLE users;

CRUD Operations

INSERT - Create Data

-- Insert single row
INSERT INTO users (email, username, password_hash, first_name, last_name)
VALUES ('ada@example.com', 'ada_lovelace', 'hashed_password', 'Ada', 'Lovelace');

-- Insert multiple rows
INSERT INTO users (email, username, password_hash, first_name, last_name) VALUES
  ('grace@example.com', 'grace_hopper', 'hashed_pwd', 'Grace', 'Hopper'),
  ('alan@example.com', 'alan_turing', 'hashed_pwd', 'Alan', 'Turing'),
  ('margaret@example.com', 'margaret_hamilton', 'hashed_pwd', 'Margaret', 'Hamilton');

-- Insert with ON DUPLICATE KEY UPDATE
INSERT INTO users (email, username, password_hash) 
VALUES ('ada@example.com', 'ada_lovelace', 'new_hash')
ON DUPLICATE KEY UPDATE password_hash = VALUES(password_hash);

SELECT - Read Data

-- Select all columns
SELECT * FROM users;

-- Select specific columns
SELECT id, email, username, created_at FROM users;

-- WHERE clause with conditions
SELECT * FROM users WHERE is_active = TRUE;
SELECT * FROM users WHERE created_at >= '2025-01-01';
SELECT * FROM users WHERE first_name LIKE 'A%';

-- Multiple conditions
SELECT * FROM users 
WHERE is_active = TRUE 
  AND created_at >= '2025-01-01'
  AND (first_name = 'Ada' OR last_name = 'Lovelace');

-- ORDER BY and LIMIT
SELECT * FROM users 
ORDER BY created_at DESC 
LIMIT 10;

-- Pagination with OFFSET
SELECT * FROM users 
ORDER BY id 
LIMIT 20 OFFSET 40;  -- Page 3 (20 per page)

-- DISTINCT values
SELECT DISTINCT status FROM orders;

-- Aggregate functions
SELECT COUNT(*) as total_users FROM users;
SELECT COUNT(*) as active_users FROM users WHERE is_active = TRUE;
SELECT AVG(total) as avg_order_value FROM orders;
SELECT SUM(total) as total_revenue FROM orders WHERE status = 'completed';
SELECT MIN(total), MAX(total), AVG(total) FROM orders;

-- GROUP BY
SELECT status, COUNT(*) as count, SUM(total) as revenue
FROM orders
GROUP BY status;

-- GROUP BY with HAVING
SELECT user_id, COUNT(*) as order_count, SUM(total) as total_spent
FROM orders
GROUP BY user_id
HAVING order_count >= 5
ORDER BY total_spent DESC;

UPDATE - Modify Data

-- Update single record
UPDATE users 
SET first_name = 'Ada Augusta', updated_at = CURRENT_TIMESTAMP
WHERE email = 'ada@example.com';

-- Update multiple records
UPDATE users 
SET is_active = FALSE
WHERE created_at < '2020-01-01';

-- Update with calculation
UPDATE orders
SET total = total * 1.10
WHERE status = 'pending' AND created_at < '2025-12-01';

-- Update with CASE
UPDATE users
SET status = CASE
  WHEN last_login < NOW() - INTERVAL 90 DAY THEN 'inactive'
  WHEN last_login < NOW() - INTERVAL 30 DAY THEN 'dormant'
  ELSE 'active'
END;

DELETE - Remove Data

-- Delete specific records
DELETE FROM users WHERE email = 'ada@example.com';

-- Delete with condition
DELETE FROM orders WHERE status = 'cancelled' AND created_at < '2024-01-01';

-- Delete all records (careful!)
DELETE FROM temp_table;

-- TRUNCATE (faster than DELETE for removing all rows)
TRUNCATE TABLE temp_table;

Joins & Relationships

-- INNER JOIN (only matching records)
SELECT u.username, u.email, o.order_number, o.total, o.created_at
FROM users u
INNER JOIN orders o ON u.id = o.user_id
WHERE o.status = 'completed'
ORDER BY o.created_at DESC;

-- LEFT JOIN (all users, even without orders)
SELECT u.username, COUNT(o.id) as order_count, COALESCE(SUM(o.total), 0) as total_spent
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
GROUP BY u.id, u.username
ORDER BY total_spent DESC;

-- RIGHT JOIN
SELECT * FROM orders o
RIGHT JOIN users u ON o.user_id = u.id;

-- Multiple joins
SELECT 
  u.username,
  o.order_number,
  oi.product_name,
  oi.quantity,
  oi.price
FROM users u
INNER JOIN orders o ON u.id = o.user_id
INNER JOIN order_items oi ON o.id = oi.order_id
WHERE u.id = 1;

-- Self join (find users in same city)
SELECT 
  u1.username as user1,
  u2.username as user2,
  u1.city
FROM users u1
INNER JOIN users u2 ON u1.city = u2.city AND u1.id < u2.id;

Indexing for Performance

-- Create single-column index
CREATE INDEX idx_email ON users(email);

-- Create unique index
CREATE UNIQUE INDEX idx_order_number ON orders(order_number);

-- Create composite index
CREATE INDEX idx_user_status_date ON orders(user_id, status, created_at);

-- Create full-text index
CREATE FULLTEXT INDEX idx_fulltext_content ON articles(title, body);

-- Full-text search
SELECT * FROM articles 
WHERE MATCH(title, body) AGAINST('MySQL database tutorial' IN NATURAL LANGUAGE MODE);

-- Show indexes on table
SHOW INDEXES FROM users;

-- Drop index
DROP INDEX idx_email ON users;

-- Analyze query performance
EXPLAIN SELECT * FROM users WHERE email = 'ada@example.com';

-- Extended analysis
EXPLAIN FORMAT=JSON SELECT * FROM orders WHERE user_id = 1;

-- Show query execution stats
SHOW PROFILES;
SET profiling = 1;
SELECT * FROM large_table WHERE indexed_column = 'value';
SHOW PROFILE;

Transactions (ACID)

-- Basic transaction
START TRANSACTION;

UPDATE accounts SET balance = balance - 100 WHERE id = 1;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;

COMMIT;

-- Transaction with rollback on error
START TRANSACTION;

INSERT INTO orders (user_id, total) VALUES (1, 99.99);
SET @order_id = LAST_INSERT_ID();

INSERT INTO order_items (order_id, product_id, quantity, price) 
VALUES (@order_id, 101, 2, 49.99);

-- If everything is OK
COMMIT;

-- If something went wrong
-- ROLLBACK;

-- Savepoints
START TRANSACTION;

INSERT INTO logs (message) VALUES ('Step 1');
SAVEPOINT step1;

INSERT INTO logs (message) VALUES ('Step 2');
SAVEPOINT step2;

-- Rollback to specific savepoint
ROLLBACK TO step2;

COMMIT;

🚀 Advanced Topics

Stored Procedures & Functions

-- Create stored procedure
DELIMITER //
CREATE PROCEDURE GetUserOrders(IN userId INT)
BEGIN
  SELECT * FROM orders WHERE user_id = userId ORDER BY created_at DESC;
END //
DELIMITER ;

-- Call procedure
CALL GetUserOrders(1);

-- Create function
DELIMITER //
CREATE FUNCTION CalculateTax(amount DECIMAL(10,2))
RETURNS DECIMAL(10,2)
DETERMINISTIC
BEGIN
  RETURN amount * 0.15;
END //
DELIMITER ;

-- Use function
SELECT order_number, total, CalculateTax(total) as tax FROM orders;

Views

-- Create view
CREATE VIEW active_users_summary AS
SELECT 
  id,
  username,
  email,
  created_at,
  (SELECT COUNT(*) FROM orders WHERE user_id = users.id) as order_count
FROM users
WHERE is_active = TRUE;

-- Query view
SELECT * FROM active_users_summary WHERE order_count > 5;

Triggers

-- Create audit trigger
DELIMITER //
CREATE TRIGGER users_after_update
AFTER UPDATE ON users
FOR EACH ROW
BEGIN
  INSERT INTO users_audit (user_id, action, old_email, new_email, changed_at)
  VALUES (NEW.id, 'UPDATE', OLD.email, NEW.email, NOW());
END //
DELIMITER ;

JSON Data Type

-- Create table with JSON column
CREATE TABLE products (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(100),
  attributes JSON
);

-- Insert JSON data
INSERT INTO products (name, attributes) VALUES
  ('Laptop', '{"brand": "Dell", "cpu": "i7", "ram": "16GB"}'),
  ('Phone', '{"brand": "Apple", "model": "iPhone 15", "storage": "256GB"}');

-- Query JSON data
SELECT name, attributes->'$.brand' as brand FROM products;
SELECT name, JSON_EXTRACT(attributes, '$.cpu') as cpu FROM products;

-- Update JSON data
UPDATE products 
SET attributes = JSON_SET(attributes, '$.price', 999.99) 
WHERE id = 1;

Replication

-- On master server
CREATE USER 'repl'@'%' IDENTIFIED BY 'password';
GRANT REPLICATION SLAVE ON *.* TO 'repl'@'%';
SHOW MASTER STATUS;

-- On slave server (in my.cnf)
server-id=2
relay-log=slave-relay-bin
read-only=1

-- Configure slave
CHANGE MASTER TO
  MASTER_HOST='master_ip',
  MASTER_USER='repl',
  MASTER_PASSWORD='password',
  MASTER_LOG_FILE='mysql-bin.000001',
  MASTER_LOG_POS=107;

START SLAVE;
SHOW SLAVE STATUS\G
