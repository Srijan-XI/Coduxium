📚 Introduction

What is Neo4j?

Neo4j is a native graph database platform designed to store, query, and manage highly connected data. Unlike traditional relational databases that use tables, Neo4j stores data as nodes and relationships, making it perfect for analyzing complex connections like social networks, recommendation engines, fraud detection, and knowledge graphs.

Key Features

🕸️ Native Graph — Built for relationships
📝 Cypher Query — Intuitive graph query language
⚡ Fast Traversals — Instant relationship queries
🎯 ACID Compliant — Full transactions
📊 Graph Algorithms — Built-in analytics
🔍 Pattern Matching — Find complex patterns
🌐 Scalable — Horizontal scaling
🎨 Browser UI — Visual query interface

Why Choose Neo4j?

  • Relationship Focus: Query connections efficiently
  • Performance: Millions of times faster for connected data
  • Flexibility: Schema-optional, adapt easily
  • Cypher: Easy-to-learn, expressive query language
  • Visualization: Built-in graph visualization tools
  • Analytics: Advanced graph algorithms included

Docker Installation

# Pull Neo4j image
docker pull neo4j:latest

# Run container
docker run \
  --name neo4j \
  -p 7474:7474 -p 7687:7687 \
  -e NEO4J_AUTH=neo4j/your_password \
  -v $HOME/neo4j/data:/data \
  neo4j:latest

# Access browser at http://localhost:7474
# Username: neo4j, Password: your_password

📖 User Guide

Cypher Basics

// Create nodes
CREATE (alice:Person {name: 'Alice', age: 30})
CREATE (bob:Person {name: 'Bob', age: 25})
CREATE (company:Company {name: 'TechCorp'})

// Create relationships
MATCH (a:Person {name: 'Alice'})
MATCH (b:Person {name: 'Bob'})
CREATE (a)-[:KNOWS]->(b)

// Create node with relationship in one query
CREATE (carol:Person {name: 'Carol'})-[:WORKS_AT]->(c:Company {name: 'StartupXYZ'})

// Return nodes
MATCH (n:Person)
RETURN n

// Return specific properties
MATCH (p:Person)
RETURN p.name, p.age

Pattern Matching

// Find all people Alice knows
MATCH (alice:Person {name: 'Alice'})-[:KNOWS]->(friend)
RETURN friend.name

// Find friends of friends
MATCH (alice:Person {name: 'Alice'})-[:KNOWS]->()-[:KNOWS]->(fof)
RETURN DISTINCT fof.name

// Find people who work at same company
MATCH (p1:Person)-[:WORKS_AT]->(c:Company)<-[:WORKS_AT]-(p2:Person)
WHERE p1 <> p2
RETURN p1.name, p2.name, c.name

// Variable length paths
MATCH path = (alice:Person {name: 'Alice'})-[:KNOWS*1..3]->(others)
RETURN path

Update Operations

// Update node property
MATCH (p:Person {name: 'Alice'})
SET p.age = 31
RETURN p

// Add new property
MATCH (p:Person {name: 'Alice'})
SET p.email = 'alice@example.com'
RETURN p

// Add label
MATCH (p:Person {name: 'Alice'})
SET p:Employee
RETURN p

// Remove property
MATCH (p:Person {name: 'Alice'})
REMOVE p.age
RETURN p

// Delete node (must delete relationships first)
MATCH (p:Person {name: 'Alice'})-[r]-()
DELETE r, p

MERGE (Create if not exists)

// MERGE node (create if doesn't exist)
MERGE (p:Person {name: 'David'})
ON CREATE SET p.created = timestamp()
ON MATCH SET p.lastSeen = timestamp()
RETURN p

// MERGE relationship
MATCH (a:Person {name: 'Alice'})
MATCH (b:Person {name: 'Bob'})
MERGE (a)-[r:KNOWS]->(b)
RETURN r

Filtering & Sorting

// WHERE clause
MATCH (p:Person)
WHERE p.age > 25
RETURN p.name, p.age

// Multiple conditions
MATCH (p:Person)
WHERE p.age > 20 AND p.age < 40
RETURN p.name

// ORDER BY
MATCH (p:Person)
RETURN p.name, p.age
ORDER BY p.age DESC

// LIMIT
MATCH (p:Person)
RETURN p.name
ORDER BY p.age DESC
LIMIT 10

// SKIP and LIMIT (pagination)
MATCH (p:Person)
RETURN p.name
ORDER BY p.name
SKIP 10 LIMIT 10

Aggregation

// COUNT
MATCH (p:Person)
RETURN COUNT(p) as totalPeople

// GROUP BY (implicit)
MATCH (p:Person)-[:WORKS_AT]->(c:Company)
RETURN c.name, COUNT(p) as employeeCount

// Multiple aggregations
MATCH (p:Person)
RETURN 
  COUNT(p) as total,
  AVG(p.age) as avgAge,
  MIN(p.age) as youngest,
  MAX(p.age) as oldest

// COLLECT (create list)
MATCH (p:Person)-[:WORKS_AT]->(c:Company {name: 'TechCorp'})
RETURN c.name, COLLECT(p.name) as employees

🚀 Advanced Topics

Shortest Path

// Find shortest path between two nodes
MATCH path = shortestPath(
  (alice:Person {name: 'Alice'})-[:KNOWS*]-(target:Person {name: 'Zoe'})
)
RETURN path

// All shortest paths
MATCH path = allShortestPaths(
  (alice:Person {name: 'Alice'})-[:KNOWS*]-(bob:Person {name: 'Bob'})
)
RETURN path

Indexes & Constraints

// Create index
CREATE INDEX person_name FOR (p:Person) ON (p.name)

// Create unique constraint
CREATE CONSTRAINT person_email FOR (p:Person) REQUIRE p.email IS UNIQUE

// Create node key
CREATE CONSTRAINT person_id FOR (p:Person) REQUIRE p.id IS NODE KEY

// List indexes
SHOW INDEXES

// List constraints
SHOW CONSTRAINTS

// Drop index
DROP INDEX person_name

Graph Algorithms

// PageRank (requires Graph Data Science library)
CALL gds.pageRank.stream('myGraph')
YIELD nodeId, score
RETURN gds.util.asNode(nodeId).name AS name, score
ORDER BY score DESC

// Community Detection (Louvain)
CALL gds.louvain.stream('myGraph')
YIELD nodeId, communityId
RETURN gds.util.asNode(nodeId).name AS name, communityId

// Centrality (Degree)
MATCH (p:Person)
RETURN p.name, size((p)-[:KNOWS]-()) as degree
ORDER BY degree DESC

Application Integration

// Node.js (neo4j-driver)
const neo4j = require('neo4j-driver');

const driver = neo4j.driver(
  'neo4j://localhost:7687',
  neo4j.auth.basic('neo4j', 'password')
);

const session = driver.session();

try {
  const result = await session.run(
    'MATCH (p:Person {name: $name}) RETURN p',
    { name: 'Alice' }
  );
  
  result.records.forEach(record => {
    console.log(record.get('p').properties);
  });
} finally {
  await session.close();
}

await driver.close();

// Python (neo4j driver)
from neo4j import GraphDatabase

driver = GraphDatabase.driver(
    "neo4j://localhost:7687",
    auth=("neo4j", "password")
)

def get_person(tx, name):
    result = tx.run("MATCH (p:Person {name: $name}) RETURN p", name=name)
    return [record["p"] for record in result]

with driver.session() as session:
    people = session.read_transaction(get_person, "Alice")
    print(people)

driver.close()

Import/Export Data

// Import CSV
LOAD CSV WITH HEADERS FROM 'file:///people.csv' AS row
CREATE (p:Person {
  name: row.name,
  age: toInteger(row.age),
  email: row.email
})

CALL apoc.export.json.all("export.json", {})

// Backup database
neo4j-admin dump --database=neo4j --to=/backups/neo4j.dump

// Restore database
neo4j-admin load --from=/backups/neo4j.dump --database=neo4j --force
