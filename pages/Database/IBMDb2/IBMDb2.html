📚 Introduction

What is IBM Db2?

IBM Db2 is a family of data management products, including database servers, developed by IBM. It's a relational database management system (RDBMS) designed to store, analyze, and retrieve data efficiently. Db2 is optimized for high performance, availability, and scalability in enterprise environments.

⚙️ Installation & Setup

Get started with IBM Db2 Community Edition or cloud deployment.

Docker Installation (Recommended for Testing)

# Pull Db2 Community Edition
docker pull ibmcom/db2

# Run container
docker run -itd \
  --name mydb2 \
  --privileged=true \
  -p 50000:50000 \
  -e LICENSE=accept \
  -e DB2INST1_PASSWORD=YourPassword \
  -e DBNAME=testdb \
  ibmcom/db2

# Check logs
docker logs -f mydb2

# Connect to container
docker exec -it mydb2 bash -c "su - db2inst1"

Joins & Relationships

-- Create departments table
CREATE TABLE departments (
    id INTEGER NOT NULL PRIMARY KEY,
    name VARCHAR(100) NOT NULL
)

-- Add foreign key
ALTER TABLE employees
ADD CONSTRAINT fk_dept
FOREIGN KEY (department_id) REFERENCES departments(id)

-- INNER JOIN
SELECT e.first_name, e.last_name, d.name as department
FROM employees e
INNER JOIN departments d ON e.department_id = d.id

-- LEFT JOIN
SELECT e.first_name, e.last_name, d.name as department
FROM employees e
LEFT JOIN departments d ON e.department_id = d.id

-- Aggregate with GROUP BY
SELECT d.name, COUNT(e.id) as employee_count, AVG(e.salary) as avg_salary
FROM departments d
LEFT JOIN employees e ON d.id = e.department_id
GROUP BY d.name

Advanced SQL

-- Common Table Expressions (CTE)
WITH high_earners AS (
    SELECT * FROM employees WHERE salary > 80000
)
SELECT first_name, last_name, salary
FROM high_earners
ORDER BY salary DESC

-- Window Functions
SELECT 
    first_name,
    last_name,
    salary,
    RANK() OVER (ORDER BY salary DESC) as salary_rank,
    ROW_NUMBER() OVER (PARTITION BY department_id ORDER BY salary DESC) as dept_rank
FROM employees

-- Recursive CTE (Organizational chart)
WITH RECURSIVE org_chart (id, name, manager_id, level) AS (
    SELECT id, first_name, manager_id, 1
    FROM employees
    WHERE manager_id IS NULL
    UNION ALL
    SELECT e.id, e.first_name, e.manager_id, o.level + 1
    FROM employees e
    INNER JOIN org_chart o ON e.manager_id = o.id
)
SELECT * FROM org_chart

Stored Procedures

-- Create stored procedure
CREATE PROCEDURE increase_salary(
    IN emp_id INTEGER,
    IN increase_pct DECIMAL(5,2)
)
LANGUAGE SQL
BEGIN
    UPDATE employees
    SET salary = salary * (1 + increase_pct / 100)
    WHERE id = emp_id;
END

-- Call procedureCALL increase_salary(1, 10.0)

-- Drop procedure
DROP PROCEDURE increase_salary

User Management

-- Create user (OS level)
# Linux: sudo useradd dbuser

-- Grant permissions
GRANT CONNECT ON DATABASE TO USER dbuser
GRANT SELECT, INSERT, UPDATE, DELETE ON TABLE employees TO USER dbuser

-- Grant all privileges
GRANT ALL PRIVILEGES ON TABLE employees TO USER dbuser

-- Revoke permissions
REVOKE SELECT ON TABLE employees FROM USER dbuser

Application Integration

// Node.js (ibm_db)
const ibmdb = require('ibm_db');

const connStr = 'DATABASE=testdb;HOSTNAME=localhost;PORT=50000;PROTOCOL=TCPIP;UID=db2inst1;PWD=password';

ibmdb.open(connStr, (err, conn) => {
  if (err) return console.error(err);
  
  conn.query('SELECT * FROM employees', (err, rows) => {
    if (err) return console.error(err);
    console.log(rows);
    conn.close();
  });
});

// Python (ibm_db)
import ibm_db

conn = ibm_db.connect("DATABASE=testdb;HOSTNAME=localhost;PORT=50000;PROTOCOL=TCPIP;UID=db2inst1;PWD=password", "", "")

stmt = ibm_db.exec_immediate(conn, "SELECT * FROM employees")
result = ibm_db.fetch_assoc(stmt)

while result:
    print(result)
    result = ibm_db.fetch_assoc(stmt)

ibm_db.close(conn)
