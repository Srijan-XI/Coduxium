📚 Introduction

What is Google BigQuery?

Google BigQuery is a fully managed, serverless data warehouse that enables scalable analysis over petabytes of data. It's a Platform as a Service (PaaS) that supports queries using ANSI SQL. BigQuery is designed to process read-only (SELECT) queries on huge datasets, making it ideal for analytics and business intelligence.

Key Features

🚀 Serverless — No infrastructure management
⚡ Fast — Scan TB in seconds
📊 Standard SQL — ANSI SQL 2011 compliant
🤖 ML Built-in — BigQuery ML for predictions
🌐 Multi-Cloud — Query across clouds
💰 Cost-Effective — Pay per query
🔄 Real-Time — Streaming inserts
🔒 Secure — Encryption at rest & transit

Why Choose BigQuery?

  • No Ops: Zero infrastructure management
  • Speed: Massively parallel query execution
  • Scalability: Automatically scales to petabytes
  • Standard SQL: Familiar query language
  • Integration: Works with Google Cloud ecosystem

⚙️ Getting Started

BigQuery is cloud-based - access through Google Cloud Console or CLI.

Setup Google Cloud Account

1
Create GCP Account

Visit cloud.google.com and sign up for free tier ($300 credit).

2
Create Project

In Google Cloud Console, create a new project.

3
Enable BigQuery API

Navigate to APIs & Services > Enable APIs > Enable BigQuery API.

4
Access BigQuery

Go to BigQuery in the navigation menu or visit console.cloud.google.com/bigquery

Install gcloud CLI (Optional)

# Install gcloud SDK
# Download from https://cloud.google.com/sdk/docs/install

# Initialize
gcloud init

# Set project
gcloud config set project YOUR_PROJECT_ID

# Run BigQuery query from CLI
bq query --use_legacy_sql=false \
  'SELECT name, COUNT(*) as count FROM `bigquery-public-data.usa_names.usa_1910_2013` GROUP BY name ORDER BY count DESC LIMIT 10'

Run First Query

-- Query public dataset
SELECT 
  name,
  SUM(number) as total
FROM `bigquery-public-data.usa_names.usa_1910_2013`
WHERE gender = 'F'
GROUP BY name
ORDER BY total DESC
LIMIT 10;

📖 User Guide

Create Dataset & Table

-- Create dataset (in Console or via query)
-- Console: Click "CREATE DATASET"

-- Create table
CREATE TABLE my_dataset.customers (
  customer_id INT64,
  first_name STRING,
  last_name STRING,
  email STRING,
  created_at TIMESTAMP
);

-- Create table from query
CREATE TABLE my_dataset.high_value_customers AS
SELECT *
FROM my_dataset.customers
WHERE total_purchases > 10000;

-- Create external table (data in Cloud Storage)
CREATE EXTERNAL TABLE my_dataset.external_data
OPTIONS (
  format = 'CSV',
  uris = ['gs://mybucket/data/*.csv'],
  skip_leading_rows = 1
);

Load Data

-- Load from Cloud Storage
LOAD DATA INTO my_dataset.customers
FROM FILES (
  format = 'CSV',
  uris = ['gs://my bucket/customers.csv']
);

-- Insert data
INSERT INTO my_dataset.customers (customer_id, first_name, last_name, email)
VALUES (1, 'John', 'Doe', 'john@example.com');

-- Insert from SELECT
INSERT INTO my_dataset.customer_backup
SELECT * FROM my_dataset.customers;

-- Streaming inserts (via API/SDK)
-- Use tabledata.insertAll API method

Query Data

-- Basic SELECT
SELECT * FROM my_dataset.customers LIMIT 100;

-- WHERE clause with functions
SELECT 
  first_name,
  last_name,
  email
FROM my_dataset.customers
WHERE DATE(created_at) = CURRENT_DATE()
  AND email LIKE '%@gmail.com';

-- Joins
SELECT 
  c.first_name,
  c.last_name,
  COUNT(o.order_id) as order_count,
  SUM(o.total_amount) as total_spent
FROM my_dataset.customers c
LEFT JOIN my_dataset.orders o ON c.customer_id = o.customer_id
GROUP BY c.customer_id, c.first_name, c.last_name
ORDER BY total_spent DESC;

Aggregations & Window Functions

-- Aggregations
SELECT 
  DATE_TRUNC(order_date, MONTH) as month,
  COUNT(*) as total_orders,
  SUM(amount) as total_revenue,
  AVG(amount) as avg_order_value
FROM my_dataset.orders
GROUP BY month
ORDER BY month DESC;

-- Window functions
SELECT 
  customer_id,
  order_date,
  amount,
  SUM(amount) OVER (
    PARTITION BY customer_id 
    ORDER BY order_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) as running_total,
  ROW_NUMBER() OVER (
    PARTITION BY customer_id 
    ORDER BY amount DESC
  ) as purchase_rank
FROM my_dataset.orders;

-- QUALIFY (filter window function results)
SELECT 
  customer_id,
  order_date,
  amount
FROM my_dataset.orders
QUALIFY ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY order_date DESC) = 1;

Arrays & Structs

-- Create table with array
CREATE TABLE my_dataset.users (
  user_id INT64,
  name STRING,
  tags ARRAY
);

-- Insert with array
INSERT INTO my_dataset.users VALUES
  (1, 'Alice', ['active', 'premium']),
  (2, 'Bob', ['active']);

-- Query array
SELECT 
  user_id,
  name,
  tag
FROM my_dataset.users,
UNNEST(tags) as tag
WHERE tag = 'premium';

-- Struct example
CREATE TABLE my_dataset.orders (
  order_id INT64,
  customer STRUCT<
    id INT64,
    name STRING,
    email STRING
  >,
  order_date DATE
);

-- Query struct
SELECT 
  order_id,
  customer.name,
  customer.email
FROM my_dataset.orders;

🚀 Advanced Topics

Partitioning & Clustering

-- Partition by date
CREATE TABLE my_dataset.events
PARTITION BY DATE(event_timestamp)
AS
SELECT * FROM my_dataset.raw_events;

-- Partition by integer range
CREATE TABLE my_dataset.customers
PARTITION BY RANGE_BUCKET(customer_id, GENERATE_ARRAY(0, 100000, 10000))
AS
SELECT * FROM my_dataset.all_customers;

-- Clustering (for better query performance)
CREATE TABLE my_dataset.events
PARTITION BY DATE(event_timestamp)
CLUSTER BY user_id, event_type
AS
SELECT * FROM my_dataset.raw_events;

BigQuery ML

-- Create linear regression model
CREATE OR REPLACE MODEL my_dataset.sales_forecast
OPTIONS(
  model_type='LINEAR_REG',
  input_label_cols=['sales']
) AS
SELECT
  date,
  temperature,
  day_of_week,
  sales
FROM my_dataset.historical_sales;

-- Make predictions
SELECT
  *
FROM ML.PREDICT(MODEL my_dataset.sales_forecast,
  (SELECT date, temperature, day_of_week FROM my_dataset.future_dates)
);

-- Evaluate model
SELECT
  *
FROM ML.EVALUATE(MODEL my_dataset.sales_forecast,
  (SELECT * FROM my_dataset.test_data)
);

Application Integration

// Node.js (@google-cloud/bigquery)
const {BigQuery} = require('@google-cloud/bigquery');

const bigquery = new BigQuery({
  projectId: 'your-project-id',
});

// Query
async function query() {
  const sqlQuery = `
    SELECT name, COUNT(*) as count
    FROM \`bigquery-public-data.usa_names.usa_1910_2013\`
    GROUP BY name
    ORDER BY count DESC
    LIMIT 10
  `;

  const [rows] = await bigquery.query(sqlQuery);
  console.log('Rows:', rows);
}

// Python (google-cloud-bigquery)
from google.cloud import bigquery

client = bigquery.Client(project='your-project-id')

query = """
    SELECT name, COUNT(*) as count
    FROM `bigquery-public-data.usa_names.usa_1910_2013`
    GROUP BY name
    ORDER BY count DESC
    LIMIT 10
"""

results = client.query(query)

for row in results:
    print(f"{row.name}: {row.count}")

Cost Optimization

-- Estimate query cost
-- Dry run to see bytes processed
bq query --use_legacy_sql=false --dry_run 'SELECT * FROM my_dataset.large_table'

-- Use partitioned tables (scan less data)
SELECT *
FROM my_dataset.events
WHERE DATE(event_timestamp) = '2024-01-15'  -- Partition filter

-- Use clustering
-- Avoid SELECT *
SELECT specific_column1, specific_column2
FROM my_dataset.table;

-- Use materialized views
CREATE MATERIALIZED VIEW my_dataset.daily_stats AS
SELECT 
  DATE(created_at) as date,
  COUNT(*) as count
FROM my_dataset.events
GROUP BY date;
