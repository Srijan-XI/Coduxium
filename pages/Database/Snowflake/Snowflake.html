📚 Introduction

What is Snowflake?

Snowflake is a cloud-native data platform delivered as a Software-as-a-Service (SaaS). It provides a data warehouse that is faster, easier to use, and more flexible than traditional offerings. Snowflake enables data storage, processing, and analytics solutions that are faster and easier to use than traditional data warehouses.

Key Features

📖 User Guide

Create Tables

-- Create table
CREATE TABLE customers (
    customer_id NUMBER AUTOINCREMENT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    email VARCHAR(100),
    created_at TIMESTAMP_NTZ DEFAULT CURRENT_TIMESTAMP()
);

-- Create table from SELECT
CREATE TABLE high_value_customers AS
SELECT * FROM customers WHERE total_purchases > 10000;

-- Show tables
SHOW TABLES;

Load Data

-- Create stage (location for files)
CREATE STAGE my_stage;

-- Upload file via SnowSQL
PUT file:///path/to/data.csv @my_stage;

-- Copy data from stage
COPY INTO customers
FROM @my_stage/data.csv
FILE_FORMAT = (TYPE = 'CSV' FIELD_OPTIONALLY_ENCLOSED_BY = '"' SKIP_HEADER = 1);

-- Load from S3
COPY INTO customers
FROM 's3://mybucket/data/'
CREDENTIALS = (AWS_KEY_ID='...' AWS_SECRET_KEY='...')
FILE_FORMAT = (TYPE = 'CSV');

-- Load from Azure Blob
COPY INTO customers
FROM 'azure://myaccount.blob.core.windows.net/container/data/'
CREDENTIALS = (AZURE_SAS_TOKEN='...')
FILE_FORMAT = (TYPE = 'CSV');

Query Data

-- Standard SELECT
SELECT * FROM customers LIMIT 10;

-- Filtering
SELECT first_name, last_name, email
FROM customers
WHERE created_at > '2024-01-01'
ORDER BY last_name;

-- Joins
SELECT c.first_name, c.last_name, o.order_date, o.total_amount
FROM customers c
JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_date >= DATEADD(day, -30, CURRENT_DATE());

-- Aggregations
SELECT 
    DATE_TRUNC('month', order_date) as month,
    COUNT(*) as order_count,
    SUM(total_amount) as revenue
FROM orders
GROUP BY month
ORDER BY month DESC;

Time Travel

-- Query data as it was 1 hour ago
SELECT * FROM customers AT(OFFSET => -3600);

-- Query data at specific timestamp
SELECT * FROM customers AT(TIMESTAMP => '2024-01-15 10:00:00'::TIMESTAMP);

-- Before statement (query before a DELETE/UPDATE)
SELECT * FROM customers BEFORE(STATEMENT => '01a3b2c4-...');

-- Undrop table (within retention period)
UNDROP TABLE customers;

-- Clone table (zero-copy clone)
CREATE TABLE customers_backup CLONE customers;

Semi-Structured Data (JSON, Parquet)

-- Create table with VARIANT column
CREATE TABLE events (
    event_id NUMBER,
    event_data VARIANT
);

-- Insert JSON
INSERT INTO events SELECT 1, PARSE_JSON('{"user": "alice", "action": "login"}');

-- Query JSON
SELECT 
    event_data:user::STRING as user,
    event_data:action::STRING as action
FROM events;

-- Flatten nested arrays
SELECT 
    value:name::STRING as product_name,
    value:price::NUMBER as price
FROM events,
LATERAL FLATTEN(input => event_data:products);

🚀 Advanced Topics

Warehouse Management

-- Create multi-cluster warehouse
CREATE WAREHOUSE analytics_wh WITH
  WAREHOUSE_SIZE = 'LARGE'
  MIN_CLUSTER_COUNT = 1
  MAX_CLUSTER_COUNT = 5
  AUTO_SUSPEND = 60
  AUTO_RESUME = TRUE;

-- Resize warehouse
ALTER WAREHOUSE dev_wh SET WAREHOUSE_SIZE = 'MEDIUM';

-- Suspend/Resume
ALTER WAREHOUSE dev_wh SUSPEND;
ALTER WAREHOUSE dev_wh RESUME;

-- Drop warehouse
DROP WAREHOUSE dev_wh;

Data Sharing

-- Create share
CREATE SHARE sales_data;

-- Grant access to database/schema
GRANT USAGE ON DATABASE sales_db TO SHARE sales_data;
GRANT USAGE ON SCHEMA sales_db.public TO SHARE sales_data;
GRANT SELECT ON TABLE sales_db.public.orders TO SHARE sales_data;

-- Add consumer account
ALTER SHARE sales_data ADD ACCOUNTS = xyz12345;

-- Consumer: Create database from share
CREATE DATABASE shared_sales FROM SHARE provider_account.sales_data;

Streams & Tasks (CDC)

-- Create stream (captures changes)
CREATE STREAM customer_stream ON TABLE customers;

-- View changes
SELECT * FROM customer_stream;

-- Create task (scheduled job)
CREATE TASK process_changes
  WAREHOUSE = dev_wh
  SCHEDULE = '5 MINUTE'
WHEN
  SYSTEM$STREAM_HAS_DATA('customer_stream')
AS
  INSERT INTO customer_history
  SELECT * FROM customer_stream;

-- Start task
ALTER TASK process_changes RESUME;

-- Stop task
ALTER TASK process_changes SUSPEND;

Application Integration

// Node.js (snowflake-sdk)
const snowflake = require('snowflake-sdk');

const connection = snowflake.createConnection({
  account: 'your_account',
  username: 'your_username',
  password: 'your_password',
  warehouse: 'dev_wh',
  database: 'my_database',
  schema: 'my_schema'
});

connection.connect((err, conn) => {
  if (err) {
    console.error('Unable to connect:', err.message);
  } else {
    connection.execute({
      sqlText: 'SELECT * FROM customers LIMIT 10',
      complete: (err, stmt, rows) => {
        console.log(rows);
      }
    });
  }
});

// Python (snowflake-connector-python)
import snowflake.connector

conn = snowflake.connector.connect(
    user='your_username',
    password='your_password',
    account='your_account',
    warehouse='dev_wh',
    database='my_database',
    schema='my_schema'
)

cur = conn.cursor()
cur.execute('SELECT * FROM customers LIMIT 10')
rows = cur.fetchall()

conn.close()

Performance Optimization

-- Clustering keys (for large tables)
ALTER TABLE large_table CLUSTER BY (date_column, category);

-- Materialized views
CREATE MATERIALIZED VIEW daily_sales AS
SELECT 
    DATE_TRUNC('day', order_date) as day,
    SUM(amount) as total_sales
FROM orders
GROUP BY day;
