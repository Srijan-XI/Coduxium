📚 Introduction

What is PostgreSQL?

PostgreSQL (often called "Postgres") is a powerful, open-source object-relational database system with over 35 years of active development. Known for its reliability, data integrity, and correctness, PostgreSQL has earned a strong reputation for architecture that has proven scalability, extensibility, and adherence to SQL standards.

Key Features

🔒 ACID Compliant — Full transaction support with data integrity
🎯 MVCC — Multi-Version Concurrency Control for consistent reads
📊 Advanced Indexing — B-tree, Hash, GiST, GIN, BRIN, Bloom filters
🔌 Extensible — Custom data types, functions, operators, and extensions
📄 JSONB Support — Efficient binary JSON storage and indexing
🌐 Standards Compliant — Excellent SQL standards conformance
🔍 Full-Text Search — Built-in text search capabilities
🗺️ Geospatial — PostGIS extension for geographic data

Why Choose PostgreSQL?

  • Data Integrity: Comprehensive support for foreign keys, check constraints, and triggers ensures data consistency.
  • Concurrency: MVCC allows readers to never block writers and vice versa, enabling high concurrency workloads.
  • Extensibility: Add custom data types, operators, index types, and procedural languages without modifying core code.
  • Rich Data Types: Arrays, hstore, JSONB, geometric types, network addresses, UUID, and more out of the box.
  • Advanced Features: Common Table Expressions (CTEs), window functions, lateral joins, materialized views, and sophisticated query planner.
  • Community & Ecosystem: Vibrant open-source community, extensive documentation, and commercial support available.

Use Cases

🌐 Web Applications — Django, Ruby on Rails, Node.js backends
📊 Data Warehousing — Analytics, OLAP, business intelligence
🗺️ GIS Applications — Location-based services with PostGIS
📱 Mobile Backends — RESTful APIs, GraphQL services
💼 Enterprise Systems — ERP, CRM, financial applications
🔬 Scientific Data — Research databases, genomics, astronomy

PostgreSQL vs Other Databases

  • vs MySQL: Better standards compliance, more advanced features (CTEs, window functions), superior handling of concurrent writes
  • vs Oracle: Open-source alternative with similar capabilities, no licensing costs, better extensibility
  • vs MongoDB: Structured data with schema enforcement, ACID transactions, but also supports JSON (JSONB)
⚠️ Important: PostgreSQL follows strict SQL standards and is case-sensitive in identifiers when quoted. Use lowercase for table/column names unless you have specific requirements.

Create Tables

-- Create users table
CREATE TABLE users (
  id BIGSERIAL PRIMARY KEY,
  email TEXT NOT NULL UNIQUE,
  username VARCHAR(50) NOT NULL UNIQUE,
  password_hash TEXT NOT NULL,
  first_name VARCHAR(100),
  last_name VARCHAR(100),
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now(),
  is_active BOOLEAN DEFAULT true
);

-- Create indexes
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_created_at ON users(created_at);

-- Create orders table with foreign key
CREATE TABLE orders (
  id BIGSERIAL PRIMARY KEY,
  user_id BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  order_number VARCHAR(50) NOT NULL UNIQUE,
  total NUMERIC(10, 2) NOT NULL,
  status VARCHAR(20) DEFAULT 'pending' 
    CHECK (status IN ('pending', 'processing', 'completed', 'cancelled')),
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_orders_user_id ON users(id);
CREATE INDEX idx_orders_status ON orders(status);
CREATE INDEX idx_orders_created_at ON orders(created_at);

-- Show table structure
\d users

-- Show all tables
\dt

-- Show table with indexes
\d+ users

CRUD Operations

INSERT - Create Data

-- Insert single row
INSERT INTO users (email, username, password_hash, first_name, last_name)
VALUES ('ada@example.com', 'ada_lovelace', 'hashed_password', 'Ada', 'Lovelace');

-- Insert and return data
INSERT INTO users (email, username, password_hash)
VALUES ('grace@example.com', 'grace', 'hashed_pwd')
RETURNING id, email, created_at;

-- Insert multiple rows
INSERT INTO users (email, username, password_hash, first_name, last_name) VALUES
  ('alan@example.com', 'alan_turing', 'hash', 'Alan', 'Turing'),
  ('margaret@example.com', 'margaret', 'hash', 'Margaret', 'Hamilton');

-- Upsert (insert or update on conflict)
INSERT INTO users (email, username, password_hash)
VALUES ('ada@example.com', 'ada_lovelace', 'new_hash')
ON CONFLICT (email) 
DO UPDATE SET password_hash = EXCLUDED.password_hash, updated_at = now();

SELECT - Read Data

-- Basic select
SELECT * FROM users;

-- Select specific columns
SELECT id, email, username, created_at FROM users;

-- WHERE conditions
SELECT * FROM users WHERE is_active = true;
SELECT * FROM users WHERE created_at >= '2025-01-01';
SELECT * FROM users WHERE first_name LIKE 'A%';
SELECT * FROM users WHERE email ILIKE '%@example.com';  -- Case-insensitive

-- Complex conditions
SELECT * FROM users 
WHERE is_active = true 
  AND created_at >= '2025-01-01'
  AND (first_name = 'Ada' OR last_name = 'Lovelace');

-- ORDER BY and LIMIT
SELECT * FROM users 
ORDER BY created_at DESC 
LIMIT 10;

-- Pagination with OFFSET
SELECT * FROM users 
ORDER BY id 
LIMIT 20 OFFSET 40;

-- DISTINCT
SELECT DISTINCT status FROM orders;

-- Aggregate functions
SELECT COUNT(*) FROM users;
SELECT COUNT(*) FILTER (WHERE is_active = true) as active_count FROM users;
SELECT AVG(total), SUM(total), MIN(total), MAX(total) FROM orders;

-- GROUP BY
SELECT status, COUNT(*) as count, SUM(total) as revenue
FROM orders
GROUP BY status;

-- GROUP BY with HAVING
SELECT user_id, COUNT(*) as order_count, SUM(total) as total_spent
FROM orders
GROUP BY user_id
HAVING COUNT(*) >= 5
ORDER BY total_spent DESC;

UPDATE - Modify Data

-- Update single record
UPDATE users 
SET first_name = 'Ada Augusta', updated_at = now()
WHERE email = 'ada@example.com';

-- Update with RETURNING
UPDATE users 
SET is_active = false
WHERE created_at < '2020-01-01'
RETURNING id, email;

-- Update with calculation
UPDATE orders
SET total = total * 1.10
WHERE status = 'pending';

-- Update with subquery
UPDATE users
SET last_order_date = (
  SELECT MAX(created_at) 
  FROM orders 
  WHERE orders.user_id = users.id
);

DELETE - Remove Data

-- Delete specific records
DELETE FROM users WHERE email = 'ada@example.com';

-- Delete with RETURNING
DELETE FROM orders 
WHERE status = 'cancelled' AND created_at < '2024-01-01'
RETURNING id, order_number;

-- Delete all records
DELETE FROM temp_table;

-- TRUNCATE (faster)
TRUNCATE TABLE temp_table;

Advanced Queries

Joins

-- INNER JOIN
SELECT u.username, u.email, o.order_number, o.total, o.created_at
FROM users u
INNER JOIN orders o ON u.id = o.user_id
WHERE o.status = 'completed'
ORDER BY o.created_at DESC;

-- LEFT JOIN
SELECT u.username, COUNT(o.id) as order_count, COALESCE(SUM(o.total), 0) as total_spent
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
GROUP BY u.id, u.username
ORDER BY total_spent DESC;

Common Table Expressions (CTEs)

-- Basic CTE
WITH active_users AS (
  SELECT id, username, email
  FROM users
  WHERE is_active = true
)
SELECT * FROM active_users WHERE username LIKE 'a%';

-- Multiple CTEs
WITH 
  recent_orders AS (
    SELECT * FROM orders WHERE created_at >= now() - interval '30 days'
  ),
  order_stats AS (
    SELECT user_id, COUNT(*) as order_count, SUM(total) as total_spent
    FROM recent_orders
    GROUP BY user_id
  )
SELECT u.username, os.order_count, os.total_spent
FROM users u
JOIN order_stats os ON u.id = os.user_id
ORDER BY os.total_spent DESC;

-- Recursive CTE (organizational hierarchy)
WITH RECURSIVE org_chart AS (
  SELECT id, name, manager_id, 1 as level
  FROM employees
  WHERE manager_id IS NULL
  
  UNION ALL
  
  SELECT e.id, e.name, e.manager_id, oc.level + 1
  FROM employees e
  JOIN org_chart oc ON e.manager_id = oc.id
)
SELECT * FROM org_chart ORDER BY level, name;

Window Functions

-- Row number
SELECT id, name, created_at,
  ROW_NUMBER() OVER (ORDER BY created_at DESC) as row_num
FROM users;

-- Rank with partitioning
SELECT 
  user_id,
  order_number,
  total,
  created_at,
  RANK() OVER (PARTITION BY user_id ORDER BY total DESC) as rank_in_user_orders
FROM orders;

-- Running total
SELECT 
  created_at::date as date,
  total,
  SUM(total) OVER (ORDER BY created_at) as running_total
FROM orders
ORDER BY created_at;

-- Moving average
SELECT 
  created_at::date as date,
  total,
  AVG(total) OVER (
    ORDER BY created_at 
    ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
  ) as moving_avg_7days
FROM orders;

Indexing Strategies

-- B-tree index (default, general purpose)
CREATE INDEX idx_users_email ON users(email);

-- Unique index
CREATE UNIQUE INDEX idx_order_number ON orders(order_number);

-- Composite index
CREATE INDEX idx_orders_user_status ON orders(user_id, status, created_at);

-- Partial index (conditional)
CREATE INDEX idx_active_users ON users(email) WHERE is_active = true;

-- Expression index
CREATE INDEX idx_lower_email ON users(LOWER(email));

-- GIN index for JSONB
CREATE INDEX idx_data_gin ON products USING GIN (attributes);

-- GiST index for full-text search
CREATE INDEX idx_articles_fts ON articles USING GiST (to_tsvector('english', content));

-- BRIN index (for very large tables with natural ordering)
CREATE INDEX idx_logs_created_brin ON logs USING BRIN (created_at);

\di

-- Analyze query performance
EXPLAIN SELECT * FROM users WHERE email = 'ada@example.com';
EXPLAIN ANALYZE SELECT * FROM users WHERE email = 'ada@example.com';

-- Drop index
DROP INDEX idx_users_email;

🚀 Advanced Features

JSONB Data Type

-- Create table with JSONB
CREATE TABLE products (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  attributes JSONB
);

-- Insert JSON data
INSERT INTO products (name, attributes) VALUES
  ('Laptop', '{"brand": "Dell", "cpu": "i7", "ram": "16GB"}'),
  ('Phone', '{"brand": "Apple", "model": "iPhone 15", "storage": "256GB"}');

-- Query JSON data
SELECT name, attributes->>'brand' as brand FROM products;
SELECT name, attributes->'cpu' as cpu FROM products;

-- Query nested JSON
SELECT name FROM products WHERE attributes @> '{"brand": "Apple"}';

-- Update JSON
UPDATE products 
SET attributes = attributes || '{"price": 999.99}'::jsonb 
WHERE id = 1;

-- Create GIN index on JSONB
CREATE INDEX idx_attributes ON products USING GIN (attributes);

Full-Text Search

-- Create table for articles
CREATE TABLE articles (
  id SERIAL PRIMARY KEY,
  title TEXT,
  content TEXT,
  search_vector tsvector
);

-- Populate search vector
UPDATE articles 
SET search_vector = to_tsvector('english', coalesce(title, '') || ' ' || coalesce(content, ''));

-- Create GIN index
CREATE INDEX idx_search_vector ON articles USING GIN (search_vector);

-- Search articles
SELECT id, title, ts_rank(search_vector, query) as rank
FROM articles, 
     to_tsquery('english', 'postgresql & database') as query
WHERE search_vector @@ query
ORDER BY rank DESC;

Views & Materialized Views

-- Create view
CREATE VIEW active_users_summary AS
SELECT 
  id,
  username,
  email,
  created_at,
  (SELECT COUNT(*) FROM orders WHERE user_id = users.id) as order_count
FROM users
WHERE is_active = true;

-- Query view
SELECT * FROM active_users_summary WHERE order_count > 5;

-- Materialized view (cached results)
CREATE MATERIALIZED VIEW monthly_sales AS
SELECT 
  date_trunc('month', created_at) as month,
  SUM(total) as total_sales,
  COUNT(*) as order_count
FROM orders
WHERE status = 'completed'
GROUP BY date_trunc('month', created_at);

-- Refresh materialized view
REFRESH MATERIALIZED VIEW monthly_sales;

-- Create index on materialized view
CREATE INDEX idx_monthly_sales_month ON monthly_sales(month);

Functions & Triggers

-- Create function
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Create trigger
CREATE TRIGGER trigger_update_users
BEFORE UPDATE ON users
FOR EACH ROW
EXECUTE FUNCTION update_updated_at();

-- Test trigger
UPDATE users SET first_name = 'Ada' WHERE id = 1;
-- updated_at will be automatically set
