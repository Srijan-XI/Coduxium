📚 Introduction

What is TypeScript?

TypeScript is a strongly-typed superset of JavaScript developed by Microsoft. It adds optional static typing, classes, and interfaces to JavaScript, compiling to clean, readable JavaScript code. TypeScript is widely used in large-scale applications, particularly with Angular, React, and Vue.

Key Features

🔒 Static Typing — Catch errors at compile-time
📝 Type Inference — Smart type detection
🎯 Interfaces — Define contracts for objects
🔧 Generics — Reusable type-safe code
🌐 ES6+ Support — Modern JavaScript features
🛠️ Tooling — Excellent IDE support
📦 Gradual Adoption — Add types incrementally
♻️ Compiles to JS — Runs anywhere JavaScript runs

Why Choose TypeScript?

  • Type Safety: Prevent runtime errors with compile-time checks
  • Better IDE Support: Autocomplete, refactoring, navigation
  • Scalability: Maintain large codebases with confidence
  • Popular: Used by Angular, Vue 3, React (optionally)
  • Community: Strong adoption in enterprise and open-source
⚠️ Note: TypeScript requires a compilation step but provides zero runtime overhead - it compiles to standard JavaScript.

⚙️ Installation & Setup

Install TypeScript

# Install globally
npm install -g typescript

# Verify installation
tsc --version

# Install in project (recommended)
npm install --save-dev typescript

Initialize TypeScript Project

# Create package.json
npm init -y

# Install TypeScript
npm install --save-dev typescript

# Create tsconfig.json
tsc --init

# Install Node types (if using Node.js)
npm install --save-dev @types/node

Basic tsconfig.json

{
  "compilerOptions": {
    "target": "ES2020",
    "module": "commonjs",
    "lib": ["ES2020"],
    "outDir": "./dist",
    "rootDir": "./src",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules"]
}

// Run

📖 User Guide

Basic Types

// Primitive types
let name: string = "Alice";
let age: number = 30;
let isActive: boolean = true;
let nothing: null = null;
let notDefined: undefined = undefined;

// Arrays
let numbers: number[] = [1, 2, 3];
let strings: Array = ["a", "b", "c"];

// Tuple
let tuple: [string, number] = ["Alice", 30];

// Any (avoid when possible)
let anything: any = "can be anything";

// Unknown (safer than any)
let value: unknown = "test";

// Void (no return value)
function log(message: string): void {
    console.log(message);
}

// Never (never returns)
function error(message: string): never {
    throw new Error(message);
}

Interfaces

// Interface definition
interface Person {
    name: string;
    age: number;
    email?: string;  // Optional property
    readonly id: string;  // Read-only
}

// Using interface
const person: Person = {
    name: "Alice",
    age: 30,
    id: "123"
};

// Function interface
interface MathFunc {
    (x: number, y: number): number;
}

const add: MathFunc = (x, y) => x + y;

// Extending interfaces
interface Employee extends Person {
    employeeId: string;
    department: string;
}

Type Aliases

// Type alias
type ID = string | number;
type Point = { x: number; y: number };

// Union types
type Status = "pending" | "approved" | "rejected";
let status: Status = "pending";

// Intersection types
type Draggable = { drag: () => void };
type Resizable = { resize: () => void };
type UIWidget = Draggable & Resizable;

// Literal types
type Direction = "north" | "south" | "east" | "west";

Functions

// Function with types
function add(a: number, b: number): number {
    return a + b;
}

// Arrow function
const multiply = (a: number, b: number): number => a * b;

// Optional parameters
function greet(name: string, greeting?: string): string {
    return `${greeting || "Hello"}, ${name}!`;
}

// Default parameters
function power(base: number, exponent: number = 2): number {
    return Math.pow(base, exponent);
}

// Rest parameters
function sum(...numbers: number[]): number {
    return numbers.reduce((a, b) => a + b, 0);
}

Classes

class Person {
    // Properties
    private name: string;
    protected age: number;
    public email: string;
    
    // Constructor
    constructor(name: string, age: number, email: string) {
        this.name = name;
        this.age = age;
        this.email = email;
    }
    
    // Method
    greet(): string {
        return `Hi, I'm ${this.name}`;
    }
    
    // Getter
    get details(): string {
        return `${this.name}, ${this.age}`;
    }
    
    // Static method
    static species(): string {
        return "Homo sapiens";
    }
}

// Inheritance
class Employee extends Person {
    constructor(
        name: string,
        age: number,
        email: string,
        public employeeId: string
    ) {
        super(name, age, email);
    }
    
    work(): string {
        return `${this.greet()} and I'm working`;
    }
}

Generics

// Generic function
function identity(arg: T): T {
    return arg;
}

let output = identity("hello");
let num = identity(42);  // Type inference

// Generic interface
interface Container {
    value: T;
    getValue(): T;
}

const stringContainer: Container = {
    value: "hello",
    getValue() {
        return this.value;
    }
};

// Generic class
class DataStore {
    private data: T[] = [];
    
    add(item: T): void {
        this. data.push(item);
    }
    
    get(index: number): T {
        return this.data[index];
    }
}

const numberStore = new DataStore();
numberStore.add(42);

🚀 Advanced Topics

Utility Types

interface User {
    id: string;
    name: string;
    email: string;
}

// Partial - all properties optional
type PartialUser = Partial;

// Required - all properties required
type RequiredUser = Required;

// Pick - select specific properties
type UserPreview = Pick;

// Omit - exclude specific properties
type UserWithoutEmail = Omit;

// Readonly - make properties read-only
type ReadonlyUser = Readonly;

Type Guards

// typeof guard
function processValue(value: string | number) {
    if (typeof value === "string") {
        return value.toUpperCase();
    }
    return value.toFixed(2);
}

// instanceof guard
class Dog {
    bark() { console.log("Woof!"); }
}

function handlePet(pet: Dog | Cat) {
    if (pet instanceof Dog) {
        pet.bark();
    }
}

// Custom type guard
interface Fish {
    swim(): void;
}

interface Bird {
    fly(): void;
}

function isFish(pet: Fish | Bird): pet is Fish {
    return (pet as Fish).swim !== undefined;
}

Decorators

// Enable in tsconfig: "experimentalDecorators": true

// Class decorator
function sealed(constructor: Function) {
    Object.seal(constructor);
    Object.seal(constructor.prototype);
}

@sealed
class MyClass {
    // ...
}

// Method decorator
function log(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    descriptor.value = function(...args: any[]) {
        console.log(`Calling ${propertyKey}`);
        return originalMethod.apply(this, args);
    };
}
