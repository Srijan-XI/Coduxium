📚 Introduction

What is Swift?

Swift is a powerful and intuitive programming language developed by Apple for iOS, macOS, watchOS, and tvOS. Introduced in 2014, Swift combines modern language features with safety and performance for building incredible apps.

Key Features

📱 iOS Development — Native Apple platform language
🛡️ Safe — Memory safety, type safety
⚡ Fast — Performance comparable to C++
🎨 SwiftUI — Declarative UI framework
📝 Modern — Clean, expressive syntax
🔄 Playgrounds — Interactive development

Why Choose Swift?

  • Apple Ecosystem: Official language for all Apple platforms
  • Modern: Designed from scratch for modern development
  • Safe: Prevents common programming errors
  • SwiftUI: Build UIs with less code

📖 User Guide

Variables & Constants

// Constants (let)
let name = "Alice"
let age = 30

// Variables (var)
var score = 0
score = 100

// Type annotation
let price: Double = 99.99
let isActive: Bool = true

// Optional
var middleName: String? = nil
middleName = "Jane"

// Optional binding
if let name = middleName {
    print("Middle name is \(name)")
}

Functions

// Basic function
func greet(name: String) -> String {
    return "Hello, \(name)!"
}

// Short form
func add(_ a: Int, _ b: Int) -> Int {
    a + b
}

// Default parameters
func power(base: Int, exponent: Int = 2) -> Int {
    Int(pow(Double(base), Double(exponent)))
}

// Multiple return values
func minMax(numbers: [Int]) -> (min: Int, max: Int) {
    (numbers.min()!, numbers.max()!)
}

print(greet(name: "Alice"))

Classes & Structs

// Struct (value type)
struct Person {
    var name: String
    var age: Int
    
    func greet() -> String {
        "Hi, I'm \(name)"
    }
}

// Class (reference type)
class Rectangle {
    var width: Double
    var height: Double
    
    init(width: Double, height: Double) {
        self.width = width
        self.height = height
    }
    
    func area() -> Double {
        width * height
    }
}

let person = Person(name: "Alice", age: 30)
let rect = Rectangle(width: 10, height: 20)

Arrays & Dictionaries

// Arrays
var fruits = ["apple", "banana", "cherry"]
fruits.append("orange")

// Dictionary
var ages = [
    "Alice": 30,
    "Bob": 25
]

ages["Charlie"] = 35

// Iteration
for fruit in fruits {
    print(fruit)
}

for (name, age) in ages {
    print("\(name) is \(age)")
}

Closures

// Closure
let add = { (a: Int, b: Int) -> Int in
    return a + b
}

// Trailing closure
let numbers = [1, 2, 3, 4, 5]

let doubled = numbers.map { $0 * 2 }
let evens = numbers.filter { $0 % 2 == 0 }
let sum = numbers.reduce(0) { $0 + $1 }

🚀 Advanced Topics

SwiftUI

import SwiftUI

struct ContentView: View {
    @State private var name = ""
    
    var body: some View {
        VStack {
            Text("Hello, \(name)!")
                .font(.title)
            
            TextField("Enter name", text: $name)
                .textFieldStyle(.roundedBorder)
                .padding()
            
            Button("Greet") {
                print("Hello, \(name)!")
            }
        }
        .padding()
    }
}

Combine (Reactive Programming)

import Combine

class ViewModel: ObservableObject {
    @Published var data: [String] = []
    
    func loadData() {
        // Async data loading
        URLSession.shared.dataTaskPublisher(for: url)
            .map { $0.data }
            .decode(type: [String].self, decoder: JSONDecoder())
            .receive(on: DispatchQueue.main)
            .assign(to: &$data)
    }
}
