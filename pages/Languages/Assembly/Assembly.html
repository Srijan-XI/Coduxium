📚 Introduction

What is Assembly Language?

Assembly Language is a low-level programming language that provides direct control over the computer's hardware. It uses human-readable mnemonics to represent machine code instructions, offering a one-to-one correspondence with the CPU's instruction set. Each CPU architecture (x86, ARM, MIPS, etc.) has its own assembly language.

Key Features

🔧 Hardware Control — Direct CPU register access
⚡ Maximum Performance — No overhead, pure speed
🎯 Precision — Control every instruction
💾 Memory Management — Manual memory control
🔍 Debugging — Understand program behavior
🛠️ System Programming — OS kernels, drivers
🔐 Security Research — Reverse engineering
🎮 Embedded Systems — IoT, microcontrollers

Why Learn Assembly?

  • Deep Understanding: Know how computers really work
  • Performance Optimization: Write critical performance sections
  • System Programming: Operating systems, bootloaders, drivers
  • Reverse Engineering: Security analysis, malware research
  • Embedded Systems: Direct hardware programming
  • Better High-Level Code: Write more efficient C/C++/Rust

Common Assembly Variants

⚠️ Note: This guide focuses on x86-64 Assembly (NASM syntax), the most common for learning and desktop development.

📖 User Guide

Hello World (x86-64 Linux)

; Hello World in NASM Assembly (Linux x86-64)
section .data
    msg db 'Hello, Coduxium XI!', 0x0A    ; Message with newline
    len equ $ - msg                        ; Calculate length

section .text
    global _start

_start:
    ; Write system call
    mov rax, 1          ; sys_write
    mov rdi, 1          ; stdout
    mov rsi, msg        ; message address
    mov rdx, len        ; message length
    syscall

    ; Exit system call
    mov rax, 60         ; sys_exit
    xor rdi, rdi        ; return 0
    syscall

Registers (x86-64)

; General Purpose Registers (64-bit)
; RAX, RBX, RCX, RDX - General purpose
; RSI, RDI - Source/Destination index
; RSP - Stack pointer
; RBP - Base pointer
; R8-R15 - Additional registers

; Example: Moving data
mov rax, 10         ; Move immediate value to RAX
mov rbx, rax        ; Copy RAX to RBX
add rax, 5          ; Add 5 to RAX (now 15)
sub rbx, 3          ; Subtract 3 from RBX (now 7)

; 32-bit, 16-bit, 8-bit access
mov eax, 100        ; 32-bit (lower half of RAX)
mov ax, 50          ; 16-bit (lower half of EAX)
mov al, 10          ; 8-bit (lower byte of AX)

Data Movement

section .data
    num1 dq 100         ; Define quadword (64-bit)
    num2 dd 50          ; Define doubleword (32-bit)
    char db 'A'         ; Define byte

section .text
    ; Load from memory
    mov rax, [num1]     ; Load 64-bit value
    mov ebx, [num2]     ; Load 32-bit value
    mov cl, [char]      ; Load byte

    ; Store to memory
    mov [num1], rax     ; Store RAX to num1
    
    ; LEA (Load Effective Address)
    lea rsi, [num1]     ; Load address of num1

Arithmetic Operations

; Addition and Subtraction
mov rax, 10
add rax, 5          ; RAX = 15
sub rax, 3          ; RAX = 12
inc rax             ; RAX = 13
dec rax             ; RAX = 12

; Multiplication
mov rax, 5
mov rbx, 3
imul rbx            ; RAX = RAX * RBX = 15

; Division (RDX:RAX / operand)
mov rax, 20
xor rdx, rdx        ; Clear RDX
mov rbx, 4
idiv rbx            ; RAX = quotient (5), RDX = remainder

Control Flow

; Conditional jumps
mov rax, 10
cmp rax, 10         ; Compare
je equal            ; Jump if equal
jne not_equal       ; Jump if not equal
jg greater          ; Jump if greater
jl less             ; Jump if less

equal:
    ; Code when equal
    jmp end

not_equal:
    ; Code when not equal

end:
    ; Continue

; Loops
mov rcx, 5          ; Counter
loop_start:
    ; Loop body
    dec rcx
    jnz loop_start  ; Jump if not zero

Functions/Procedures

; Function definition
add_numbers:
    ; RDI = first parameter
    ; RSI = second parameter
    mov rax, rdi
    add rax, rsi
    ret             ; Return (result in RAX)

; Calling convention (System V AMD64)
main:
    mov rdi, 10     ; First argument
    mov rsi, 20     ; Second argument
    call add_numbers
    ; Result now in RAX (30)

Stack Operations

; Push and Pop
push rax            ; Save RAX on stack
push rbx            ; Save RBX on stack
; ... do work ...
pop rbx             ; Restore RBX
pop rax             ; Restore RAX

; Function prologue/epilogue
my_function:
    push rbp        ; Save base pointer
    mov rbp, rsp    ; Set up stack frame
    sub rsp, 16     ; Allocate local space
    
    ; Function body
    
    mov rsp, rbp    ; Clean up
    pop rbp         ; Restore base pointer
    ret

🚀 Advanced Topics

System Calls (Linux x86-64)

; Common system calls
; RAX = syscall number
; Arguments in: RDI, RSI, RDX, R10, R8, R9

; Read from stdin
mov rax, 0          ; sys_read
mov rdi, 0          ; stdin
mov rsi, buffer     ; destination buffer
mov rdx, 100        ; max bytes
syscall

; Write to stdout
mov rax, 1          ; sys_write
mov rdi, 1          ; stdout
mov rsi, msg        ; source buffer
mov rdx, len        ; length
syscall

; Open file
mov rax, 2          ; sys_open
mov rdi, filename   ; file path
mov rsi, 0          ; O_RDONLY
syscall
mov r8, rax         ; Save file descriptor

Bitwise Operations

; AND, OR, XOR, NOT
mov rax, 0b1100
and rax, 0b1010     ; RAX = 0b1000
or rax, 0b0011      ; RAX = 0b1011
xor rax, 0b1111     ; RAX = 0b0100
not rax             ; Bitwise NOT

; Shifts
mov rax, 8
shl rax, 2          ; Shift left 2 (RAX = 32)
shr rax, 1          ; Shift right 1 (RAX = 16)

; Rotate
rol rax, 1          ; Rotate left
ror rax, 1          ; Rotate right

SIMD Instructions (SSE/AVX)

; SSE example - parallel operations
section .data
    align 16
    vec1 dd 1.0, 2.0, 3.0, 4.0
    vec2 dd 5.0, 6.0, 7.0, 8.0

section .text
    movaps xmm0, [vec1]     ; Load 4 floats
    movaps xmm1, [vec2]     ; Load 4 floats
    addps xmm0, xmm1        ; Parallel add
    movaps [result], xmm0   ; Store result

Inline Assembly in C

// GCC inline assembly
#include 

int main() {
    long result;
    long a = 10, b = 20;
    
    __asm__ (
        "mov %1, %0\n\t"
        "add %2, %0"
        : "=r" (result)
        : "r" (a), "r" (b)
    );
    
    printf("Result: %ld\n", result);
    return 0;
}
