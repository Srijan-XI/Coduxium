📚 Introduction

What is Kotlin?

Kotlin is a modern, statically-typed programming language that runs on the JVM. Developed by JetBrains and officially supported by Google for Android, Kotlin offers concise syntax, null safety, and full Java interoperability.

Key Features

📱 Android First — Official Android language
🛡️ Null Safety — No more NullPointerException
📝 Concise — Less boilerplate than Java
🔄 Coroutines — Simple async programming
♻️ Java Compatible — Use Java libraries
🎯 Modern — Functional programming support

Why Choose Kotlin?

  • Android Development: Google's preferred language
  • Safety: Null safety prevents common bugs
  • Productivity: Write less code, do more
  • Interoperable: Use existing Java code

⚙️ Installation & Setup

Install Kotlin

# Using SDKMAN
sdk install kotlin

# macOS
brew install kotlin

# Verify
kotlin -version

Android Studio (Recommended)

Download Android Studio from developer.android.com/studio - Kotlin support is built-in.

Your First Kotlin Program

// Main.kt
fun main() {
    println("Hello, Kotlin!")
}

// Compile and run
kotlinc Main.kt -include-runtime -d main.jar
java -jar main.jar

📖 User Guide

Variables & Types

fun main() {
    // Immutable (val)
    val name: String name = "Alice"
    val age = 30  // Type inference
    
    // Mutable (var)
    var count = 0
    count = 1
    
    // Null safety
    var nullable: String? = null
    nullable = "value"
    
    // Safe call
    println(nullable?.length)
    
    // Elvis operator
    val length = nullable?.length ?: 0
}

Functions

// Basic function
fun greet(name: String): String {
    return "Hello, $name!"
}

// Expression body
fun add(a: Int, b: Int) = a + b

// Default parameters
fun power(base: Int, exponent: Int = 2) = 
    Math.pow(base.toDouble(), exponent.toDouble())

// Named arguments
greet(name = "Alice")

// Extension function
fun String.exclaim() = "$this!"
println("Hello".exclaim())  // Hello!

Classes & Objects

// Data class
data class Person(
    val name: String,
    val age: Int
)

// Regular class
class Rectangle(val width: Int, val height: Int) {
    fun area() = width * height
}

// Object (singleton)
object Config {
    const val API_KEY = "abc123"
}

// Usage
val person = Person("Alice", 30)
val rect = Rectangle(10, 20)
println(rect.area())

Collections

// Lists
val numbers = listOf(1, 2, 3, 4, 5)
val mutableList = mutableListOf("a", "b", "c")

// Map
val map = mapOf(
    "name" to "Alice",
    "age" to 30
)

// Filter
val evens = numbers.filter { it % 2 == 0 }

// Map
val squared = numbers.map { it * it }

// Reduce
val sum = numbers.reduce { acc, n -> acc + n }

When Expression

fun describe(x: Any) = when (x) {
    1 -> "One"
    in 2..10 -> "Between 2 and 10"
    is String -> "It's a string"
    else -> "Something else"
}

// As statement
when (score) {
    in 90..100 -> println("A")
    in 80..89 -> println("B")
    else -> println("C")
}

🚀 Advanced Topics

Coroutines

import kotlinx.coroutines.*

fun main() = runBlocking {
    launch {
        delay(1000L)
        println("World!")
    }
    println("Hello,")
}

// Async/await
suspend fun fetchData(): String {
    delay(1000)
    return "Data"
}

val result = async { fetchData() }
println(result.await())

Android Development

// Jetpack Compose
@Composable
fun Greeting(name: String) {
    Text(text = "Hello, $name!")
}

// ViewModel
class MyViewModel : ViewModel() {
    private val _data = MutableLiveData()
    val data: LiveData = _data
    
    fun loadData() {
        viewModelScope.launch {
            _data.value = fetchData()
        }
    }
}
