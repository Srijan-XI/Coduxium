📚 Introduction

What is Electron?

Electron is an open-source framework created by GitHub that allows you to build cross-platform desktop applications using web technologies (JavaScript, HTML, and CSS). It combines Chromium and Node.js into a single runtime, enabling you to write desktop applications that work on Windows, macOS, and Linux from a single codebase.

Key Features

🌐 Cross-Platform — Windows, macOS, Linux
🔧 Web Technologies — HTML, CSS, JavaScript
📦 Node.js Integration — Access to npm ecosystem
🖥️ Native APIs — System tray, notifications, menus
🔄 Auto Updates — Built-in update mechanism
🛡️ Chromium Security — Sandboxing, CSP
🎨 Custom UI — Full control over appearance
📱 Native Feel — Platform-specific behaviors

Why Choose Electron?

  • Single Codebase: Write once, deploy to all platforms
  • Web Skills: Use existing web development knowledge
  • Rich Ecosystem: Access to npm and web frameworks
  • Rapid Development: Fast prototyping and iteration
  • Native Integration: Access file system, hardware, OS features
  • Battle-Tested: Used by major companies worldwide

⚙️ Installation & Setup

Get started with Electron by creating your first application.

Requirements

  • Node.js (v16 or higher recommended)
  • npm or yarn

Quick Start

Method 1: electron-quick-start

# Clone the quick start repository
git clone https://github.com/electron/electron-quick-start
cd electron-quick-start

# Install dependencies
npm install

# Run the app
npm start

Method 2: Create from Scratch

# Create project directory
mkdir my-electron-app
cd my-electron-app

# Initialize npm project
npm init -y

# Install Electron
npm install --save-dev electron

Basic Project Structure

my-electron-app/
├── main.js              # Main process
├── preload.js           # Preload script
├── renderer.js          # Renderer process
├── index.html           # UI
├── package.json
└── node_modules/

Configure package.json

📖 User Guide

Main Process (main.js)

const { app, BrowserWindow } = require('electron');
const path = require('path');

let mainWindow;

function createWindow() {
  mainWindow = new BrowserWindow({
    width: 800,
    height: 600,
    webPreferences: {
      preload: path.join(__dirname, 'preload.js'),
      nodeIntegration: false,
      contextIsolation: true
    }
  });

  mainWindow.loadFile('index.html');

  // Open DevTools in development
  // mainWindow.webContents.openDevTools();

  mainWindow.on('closed', () => {
    mainWindow = null;
  });
}

app.whenReady().then(() => {
  createWindow();

  app.on('activate', () => {
    if (BrowserWindow.getAllWindows().length === 0) {
      createWindow();
    }
  });
});

app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit();
  }
});

Preload Script (preload.js)

const { contextBridge, ipcRenderer } = require('electron');

// Expose protected methods to renderer process
contextBridge.exposeInMainWorld('electronAPI', {
  getVersion: () => process.versions.electron,
  sendMessage: (message) => ipcRenderer.send('message', message),
  onReply: (callback) => ipcRenderer.on('reply', callback)
});

Renderer Process (renderer.js)

// Access exposed API from preload
document.getElementById('myBtn').addEventListener('click', () => {
  const version = window.electronAPI.getVersion();
  document.getElementById('info').textContent = 
    `Electron version: ${version}`;
});

IPC Communication (Inter-Process)

// Main process (main.js)
const { ipcMain } = require('electron');

ipcMain.on('message', (event, arg) => {
  console.log('Received:', arg);
  event.reply('reply', 'Message received!');
});

// Invoke/handle pattern (async)
ipcMain.handle('get-data', async (event, arg) => {
  const result = await fetchData();
  return result;
});

// Preload (preload.js)
contextBridge.exposeInMainWorld('electronAPI', {
  sendMessage: (msg) => ipcRenderer.send('message', msg),
  onReply: (callback) => ipcRenderer.on('reply', (event, ...args) => 
    callback(...args)
  ),
  getData: (arg) => ipcRenderer.invoke('get-data', arg)
});

// Renderer (renderer.js)
// Send message
window.electronAPI.sendMessage('Hello from renderer!');

// Listen for reply
window.electronAPI.onReply((reply) => {
  console.log(reply);
});

// Invoke with promise
window.electronAPI.getData('param').then(data => {
  console.log(data);
});

System Tray

const { Tray, Menu } = require('electron');

let tray;

function createTray() {
  tray = new Tray('path/to/icon.png');
  
  const contextMenu = Menu.buildFromTemplate([
    { label: 'Show App', click: () => mainWindow.show() },
    { label: 'Quit', click: () => app.quit() }
  ]);
  
  tray.setToolTip('My Electron App');
  tray.setContextMenu(contextMenu);
  
  tray.on('click', () => {
    mainWindow.isVisible() ? mainWindow.hide() : mainWindow.show();
  });
}

File Dialogs

const { dialog } = require('electron');

// Open file dialog
async function openFile() {
  const { canceled, filePaths } = await dialog.showOpenDialog({
    properties: ['openFile', 'multiSelections'],
    filters: [
      { name: 'Images', extensions: ['jpg', 'png', 'gif'] },
      { name: 'All Files', extensions: ['*'] }
    ]
  });
  
  if (!canceled) {
    return filePaths;
  }
}

// Save file dialog
async function saveFile() {
  const { canceled, filePath } = await dialog.showSaveDialog({
    defaultPath: 'untitled.txt'
  });
  
  if (!canceled) {
    return filePath;
  }
}

Auto Updates (electron-updater)

// Install
npm install electron-updater

// Main process
const { autoUpdater } = require('electron-updater');

autoUpdater.checkForUpdatesAndNotify();

autoUpdater.on('update-available', () => {
  console.log('Update available');
});

autoUpdater.on('update-downloaded', () => {
  autoUpdater.quitAndInstall();
});

Security Best Practices

// 1. Enable Context Isolation
webPreferences: {
  contextIsolation: true,
  nodeIntegration: false
}

// 2. Use CSP
<meta http-equiv="Content-Security-Policy" 
      content="default-src 'self'; script-src 'self'">

// 3. Validate all user input
// 4. Use HTTPS for remote content
// 5. Disable Node.js in renderer
// 6. Enable sandbox
webPreferences: {
  sandbox: true
}
