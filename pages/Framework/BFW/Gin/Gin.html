📚 Introduction

What is Gin?

Gin is a HTTP web framework written in Go (Golang). It features a Martini-like API with much better performance -- up to 40 times faster. If you need performance and good productivity, Gin is your framework of choice.

Key Features

⚡ Fast — Radix tree routing, low overhead
🛣️ Router — Fast, flexible HTTP routing
🔧 Middleware — Extensible middleware support
📦 JSON Validation — Built-in validation
🎯 Error Management — Convenient error handling
🔄 Route Grouping — Organize routes logically
📝 Rendering — JSON, XML, HTML rendering
🧪 Testable — Easy to unit test

Why Choose Gin?

  • Performance: Extremely fast, optimized routing
  • Simplicity: Clean, intuitive API
  • Productivity: Middleware, grouping, easy validation
  • Go Ecosystem: Access to all Go libraries
  • Active Community: Popular framework with good support
  • Production Ready: Used by many companies
⚠️ Requirement: Gin requires Go 1.16 or higher.

⚙️ Installation & Setup

Get started with Gin by installing the package.

Install Go

# Download from https://go.dev/dl/

# Verify installation
go version

Create New Project

# Create project directory
mkdir my-gin-app
cd my-gin-app

# Initialize Go module
go mod init my-gin-app

# Install Gin
go get -u github.com/gin-gonic/gin

Create First Application

// main.go
package main

import (
    "github.com/gin-gonic/gin"
    "net/http"
)

func main() {
    // Create router
    r := gin.Default()
    
    // Define route
    r.GET("/", func(c *gin.Context) {
        c.JSON(http.StatusOK, gin.H{
            "message": "Hello, Coduxium XI!",
        })
    })
    
    // Run server
    r.Run(":8080") // Listen on localhost:8080
}

Run the Application

# Run
go run main.go

# Or build and run
go build
./my-gin-app

# Server will run on http://localhost:8080

Basic Routing

package main

import (
    "github.com/gin-gonic/gin"
    "net/http"
)

func main() {
    r := gin.Default()
    
    // GET
    r.GET("/users", getUsers)
    
    // POST
    r.POST("/users", createUser)
    
    // PUT
    r.PUT("/users/:id", updateUser)
    
    // DELETE
    r.DELETE("/users/:id", deleteUser)
    
    r.Run(":8080")
}

func getUsers(c *gin.Context) {
    c.JSON(http.StatusOK, gin.H{
        "users": []string{"Alice", "Bob"},
    })
}

Path Parameters

// Route with parameter
r.GET("/users/:id", func(c *gin.Context) {
    id := c.Param("id")
    c.JSON(http.StatusOK, gin.H{
        "user_id": id,
    })
})

// Multiple parameters
r.GET("/users/:id/posts/:postId", func(c *gin.Context) {
    userId := c.Param("id")
    postId := c.Param("postId")
    c.JSON(http.StatusOK, gin.H{
        "user_id": userId,
        "post_id": postId,
    })
})

Query Parameters

// GET /search?q=golang&limit=10
r.GET("/search", func(c *gin.Context) {
    query := c.Query("q")
    limit := c.DefaultQuery("limit", "20") // Default value
    
    c.JSON(http.StatusOK, gin.H{
        "query": query,
        "limit": limit,
    })
})

Request Body & Binding

type User struct {
    Name  string `json:"name" binding:"required"`
    Email string `json:"email" binding:"required,email"`
    Age   int    `json:"age" binding:"gte=0,lte=120"`
}

r.POST("/users", func(c *gin.Context) {
    var user User
    
    // Bind and validate JSON
    if err := c.ShouldBindJSON(&user); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{
            "error": err.Error(),
        })
        return
    }
    
    // Process user
    c.JSON(http.StatusCreated, user)
})

Route Groups

func main() {
    r := gin.Default()
    
    // API v1 group
    v1 := r.Group("/api/v1")
    {
        v1.GET("/users", getUsers)
        v1.POST("/users", createUser)
        
        // Nested groups
        admin := v1.Group("/admin")
        {
            admin.GET("/stats", getStats)
        }
    }
    
    // API v2 group
    v2 := r.Group("/api/v2")
    {
        v2.GET("/users", getUsersV2)
    }
    
    r.Run(":8080")
}

Middleware

// Logger middleware
func Logger() gin.HandlerFunc {
    return func(c *gin.Context) {
        // Before request
        t := time.Now()
        
        // Process request
        c.Next()
        
        // After request
        latency := time.Since(t)
        log.Print(latency)
    }
}

// Use middleware
r := gin.New()
r.Use(gin.Logger())
r.Use(gin.Recovery())
r.Use(Logger())

// Per-route middleware
r.GET("/auth", AuthRequired(), func(c *gin.Context) {
    c.JSON(http.StatusOK, gin.H{"message": "Authenticated"})
})

// Group middleware
authorized := r.Group("/admin")
authorized.Use(AuthRequired())
{
    authorized.GET("/dashboard", dashboard)
}

Custom Validation

import (
    "github.com/go-playground/validator/v10"
)

type User struct {
    Name string `json:"name" binding:"required,min=3"`
    Age  int    `json:"age" binding:"required,gte=18"`
}

// Custom validator
var bookableDate validator.Func = func(fl validator.FieldLevel) bool {
    date, ok := fl.Field().Interface().(time.Time)
    if ok {
        if date.After(time.Now()) {
            return true
        }
    }
    return false
}

// Register custom validator
if v, ok := binding.Validator.Engine().(*validator.Validate); ok {
    v.RegisterValidation("bookabledate", bookableDate)
}

File Upload

r.POST("/upload", func(c *gin.Context) {
    // Single file
    file, _ := c.FormFile("file")
    
    // Save file
    c.SaveUploadedFile(file, "./uploads/"+file.Filename)
    
    c.JSON(http.StatusOK, gin.H{
        "message": "File uploaded",
        "filename": file.Filename,
    })
})

// Multiple files
r.POST("/upload-multiple", func(c *gin.Context) {
    form, _ := c.MultipartForm()
    files := form.File["files"]
    
    for _, file := range files {
        c.SaveUploadedFile(file, "./uploads/"+file.Filename)
    }
    
    c.JSON(http.StatusOK, gin.H{
        "count": len(files),
    })
})

Database Integration (GORM)

import (
    "gorm.io/driver/sqlite"
    "gorm.io/gorm"
)

type User struct {
    ID    uint   `gorm:"primaryKey"`
    Name  string
    Email string `gorm:"unique"`
}

var db *gorm.DB

func initDB() {
    var err error
    db, err = gorm.Open(sqlite.Open("test.db"), &gorm.Config{})
    if err != nil {
        panic("failed to connect database")
    }
    
    db.AutoMigrate(&User{})
}

func getUsers(c *gin.Context) {
    var users []User
    db.Find(&users)
    c.JSON(http.StatusOK, users)
}

func createUser(c *gin.Context) {
    var user User
    if err := c.ShouldBindJSON(&user); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }
    
    db.Create(&user)
    c.JSON(http.StatusCreated, user)
}

Authentication (JWT)

import (
    "github.com/golang-jwt/jwt/v5"
)

var jwtSecret = []byte("your-secret-key")

func generateToken(userID uint) (string, error) {
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, jwt.MapClaims{
        "user_id": userID,
        "exp": time.Now().Add(time.Hour * 24).Unix(),
    })
    
    return token.SignedString(jwtSecret)
}

func AuthMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        tokenString := c.GetHeader("Authorization")
        
        token, err := jwt.Parse(tokenString, func(token *jwt.Token) (interface{}, error) {
            return jwtSecret, nil
        })
        
        if err != nil || !token.Valid {
            c.JSON(http.StatusUnauthorized, gin.H{"error": "Unauthorized"})
            c.Abort()
            return
        }
        
        c.Next()
    }
}
