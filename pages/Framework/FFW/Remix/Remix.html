📚 Introduction

What is Remix?

Remix is a full-stack web framework built by the creators of React Router. It focuses on web fundamentals, progressive enhancement, and modern user experiences. Remix enables you to build fast, slick, and resilient user experiences by leveraging the platform and embracing web standards.

Key Features

⚡ Fast by Default — Optimized data loading
🌐 Web Standards — Forms, URLs, HTTP
📦 Nested Routing — Parallel data loading
🔄 Progressive Enhancement — Works without JS
🎯 Type-Safe — Full TypeScript support
🚀 SSR + Hydration — Server-side rendering
📡 Optimistic UI — Instant user feedback
🛡️ Error Handling — Built-in error boundaries

Why Choose Remix?

  • Web Fundamentals: Leverage HTML, HTTP, and platform APIs
  • Fast UX: Optimistic UI, parallel loading, prefetching
  • Simple Data Flow: Loaders and actions handle all data
  • Resilient Apps: Works even if JavaScript fails
  • React Ecosystem: Use all your favorite React libraries
  • Deploy Anywhere: Adapters for Vercel, Netlify, Cloudflare, Node
⚠️ Note: Remix v2 brought improvements to routing and data loading. Always use the latest version.

⚙️ Installation & Setup

Get started with Remix by creating a new project.

Create New Project

# Create new Remix app
npx create-remix@latest my-remix-app

# Navigate to project
cd my-remix-app

# Install dependencies
npm install

# Start development server
npm run dev

Choose Your Deployment Target

During setup, you'll be prompted to choose where to deploy:

  • Remix App Server: Simple Node.js server
  • Vercel: Serverless deployment
  • Netlify: Edge functions
  • Cloudflare Workers: Edge deployment
💡 Note: Development server runs on http://localhost:5173 (Vite) or http://localhost:3000

Routes & Loaders

// app/routes/_index.tsx
import { json } from "@remix-run/node";
import { useLoaderData } from "@remix-run/react";

export const loader = async () => {
  return json({
    message: "Hello from Remix!",
    timestamp: new Date().toISOString()
  });
};

export default function Index() {
  const data = useLoaderData<typeof loader>();
  
  return (
    <div>
      <h1>{data.message}</h1>
      <p>Time: {data.timestamp}</p>
    </div>
  );
}

Actions & Forms

// app/routes/login.tsx
import { json, redirect } from "@remix-run/node";
import { Form, useActionData } from "@remix-run/react";
import type { ActionFunctionArgs } from "@remix-run/node";

export const action = async ({ request }: ActionFunctionArgs) => {
  const formData = await request.formData();
  const email = formData.get("email");
  const password = formData.get("password");
  
  // Validate
  if (!email || !password) {
    return json({ error: "Email and password required" }, { status: 400 });
  }
  
  // Authenticate user
  // ... your auth logic
  
  return redirect("/dashboard");
};

export default function Login() {
  const actionData = useActionData<typeof action>();
  
  return (
    <Form method="post">
      <input type="email" name="email" required />
      <input type="password" name="password" required />
      {actionData?.error && <p>{actionData.error}</p>}
      <button type="submit">Login</button>
    </Form>
  );
}

Dynamic Routes

// app/routes/blog.$slug.tsx
import { json } from "@remix-run/node";
import { useLoaderData } from "@remix-run/react";
import type { LoaderFunctionArgs } from "@remix-run/node";

export const loader = async ({ params }: LoaderFunctionArgs) => {
  const post = await getPost(params.slug);
  
  if (!post) {
    throw new Response("Not Found", { status: 404 });
  }
  
  return json({ post });
};

export default function BlogPost() {
  const { post } = useLoaderData<typeof loader>();
  
  return (
    <article>
      <h1>{post.title}</h1>
      <p>{post.content}</p>
    </article>
  );
}

Error Handling

// Route-level error boundary
export function ErrorBoundary() {
  const error = useRouteError();
  
  if (isRouteErrorResponse(error)) {
    return (
      <div>
        <h1>{error.status} {error.statusText}</h1>
        <p>{error.data}</p>
      </div>
    );
  }
  
  return (
    <div>
      <h1>Error!</h1>
      <p>{error.message}</p>
    </div>
  );
}

Sessions & Cookies

// app/sessions.ts
import { createCookieSessionStorage } from "@remix-run/node";

export const { getSession, commitSession, destroySession } =
  createCookieSessionStorage({
    cookie: {
      name: "__session",
      secrets: [process.env.SESSION_SECRET],
      sameSite: "lax",
      path: "/",
      httpOnly: true,
      secure: process.env.NODE_ENV === "production"
    }
  });

// Usage in action
export const action = async ({ request }: ActionFunctionArgs) => {
  const session = await getSession(request.headers.get("Cookie"));
  session.set("userId", user.id);
  
  return redirect("/dashboard", {
    headers: {
      "Set-Cookie": await commitSession(session)
    }
  });
};

Resource Routes (API Endpoints)

// app/routes/api.users.ts
import { json } from "@remix-run/node";

export const loader = async () => {
  const users = await db.user.findMany();
  return json(users);
};

export const action = async ({ request }: ActionFunctionArgs) => {
  const body = await request.json();
  const user = await db.user.create({ data: body });
  return json(user, { status: 201 });
};
