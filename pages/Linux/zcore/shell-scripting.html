🔀 Conditionals

If Statement

if [ $AGE -gt 18 ]; then
    echo "Adult"
elif [ $AGE -eq 18 ]; then
    echo "Just turned 18"
else
    echo "Minor"
fi

String Comparison

if [ "$NAME" = "John" ]; then
    echo "Hello John"
fi

if [ -z "$VAR" ]; then
    echo "Variable is empty"
fi

File Tests

if [ -f "file.txt" ]; then
    echo "File exists"
fi

if [ -d "directory" ]; then
    echo "Directory exists"
fi

Logical Operators

if [ $AGE -gt 18 ] && [ $AGE -lt 65 ]; then
    echo "Working age"
fi

if [ $DAY = "Sat" ] || [ $DAY = "Sun" ]; then
    echo "Weekend"
fi

🔁 Loops

For Loop

for i in 1 2 3 4 5; do
    echo "Number: $i"
done

for file in *.txt; do
    echo "Processing $file"
done

C-Style For Loop

for ((i=1; i<=5; i++)); do
    echo "Count: $i"
done

While Loop

counter=1
while [ $counter -le 5 ]; do
    echo "Counter: $counter"
    ((counter++))
done

Until Loop

counter=1
until [ $counter -gt 5 ]; do
    echo "Counter: $counter"
    ((counter++))
done

Function Examples

Basic Function

Return Values

add() {
    local result=$(($1 + $2))
    echo $result
}

Local Variables

my_function() {
    local LOCAL_VAR="local"
    GLOBAL_VAR="global"
}

my_function

📚 Arrays

Create Arrays

fruits=("apple" "banana" "cherry")
numbers=(1 2 3 4 5)

Access Elements

echo ${fruits[0]}   # apple
echo ${fruits[@]}   # all elements
echo ${#fruits[@]}  # array length

Loop Through Array

for fruit in "${fruits[@]}"; do
    echo "Fruit: $fruit"
done

Add/Remove Elements

fruits+=("orange")   # add
unset fruits[1]      # remove

📁 File Operations

Read File Line by Line

while IFS= read -r line; do
    echo "Line: $line"
done < file.txt

Write to File

echo "text" > file.txt   # overwrite
echo "text" >> file.txt  # append

Check File Existence

if [ -f "file.txt" ]; then
    echo "File exists"
else
    echo "File not found"
fi

💼 Real-World Examples

System Backup Script

#!/bin/bash
# Backup script with error handling

set -e
set -u

BACKUP_DIR="/backup"
SOURCE_DIR="/home/user/documents"
DATE=$(date +%Y%m%d)
BACKUP_FILE="backup_${DATE}.tar.gz"

echo "Starting backup..."

if [ ! -d "$BACKUP_DIR" ]; then
    mkdir -p "$BACKUP_DIR"
fi

tar -czf "${BACKUP_DIR}/${BACKUP_FILE}" "$SOURCE_DIR"

if [ $? -eq 0 ]; then
    echo "Backup completed: ${BACKUP_FILE}"
else
    echo "Backup failed!" >&2
    exit 1
fi

# Clean old backups (keep last 7 days)
find "$BACKUP_DIR" -name "backup_*.tar.gz" -mtime +7 -delete

echo "Backup process finished!"

Log File Monitor

#!/bin/bash
# Monitor log file for errors

LOG_FILE="/var/log/application.log"
ERROR_COUNT=0

tail -f "$LOG_FILE" | while read -r line; do
    if echo "$line" | grep -q "ERROR"; then
        ((ERROR_COUNT++))
        echo "$(date): Error detected - $line"
        
        if [ $ERROR_COUNT -ge 5 ]; then
            echo "Too many errors! Sending alert..."
            # Send email or notification
            break
        fi
    fi
done
