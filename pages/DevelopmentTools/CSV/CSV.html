📄 Overview

CSV (Comma-Separated Values) is a simple file format used to store tabular data. Each line represents a row, and columns are separated by commas. It's widely used for data exchange between applications, especially spreadsheets and databases.

Common Use Cases

  • Data Export/Import — Database tables, spreadsheets
  • Data Analytics — Machine learning datasets
  • Reporting — Generate reports from applications
  • Data Migration — Transfer between systems

📖 CSV Format

Basic Structure

name,age,city
John Doe,30,New York
Jane Smith,25,Los Angeles
Bob Johnson,35,Chicago

With Quotes

name,description,price
"MacBook Pro","15-inch, 512GB SSD",2399.99
"iPhone 15","Latest model, 256GB",999.99
"iPad Air","10.9-inch display",599.00

Handling Special Characters

name,address,notes
"John Doe","123 Main St, Apt 4B","VIP customer"
"Jane Smith","456 Oak Ave","Prefers email, not phone"
"Bob's Store","789 Pine Rd","Uses quote: ""Best prices"""

Different Delimiters

# Semicolon-separated (common in Europe)
name;age;city
John Doe;30;New York

# Tab-separated (TSV)
name	age	city
John Doe	30	New York

# Pipe-separated
name|age|city
John Doe|30|New York

🔧 Working with CSV

JavaScript

// Read CSV (Node.js)
const fs = require('fs');
const csv = require('csv-parser');

fs.createReadStream('data.csv')
  .pipe(csv())
  .on('data', (row) => {
    console.log(row);
  });

// Write CSV
const createCsvWriter = require('csv-writer').createObjectCsvWriter;

const csvWriter = createCsvWriter({
  path: 'output.csv',
  header: [
    {id: 'name', title: 'Name'},
    {id: 'age', title: 'Age'}
  ]
});

csvWriter.writeRecords([
  {name: 'John', age: 30},
  {name: 'Jane', age: 25}
]);

// Browser - Parse CSV string
function parseCSV(text) {
  const lines = text.split('\n');
  const headers = lines[0].split(',');
  
  return lines.slice(1).map(line => {
    const values = line.split(',');
    return headers.reduce((obj, header, i) => {
      obj[header] = values[i];
      return obj;
    }, {});
  });
}

Python

import csv

# Read CSV
with open('data.csv', 'r') as file:
    reader = csv.DictReader(file)
    for row in reader:
        print(row['name'], row['age'])

# Write CSV
with open('output.csv', 'w', newline='') as file:
    fieldnames = ['name', 'age', 'city']
    writer = csv.DictWriter(file, fieldnames=fieldnames)
    
    writer.writeheader()
    writer.writerow({'name': 'John', 'age': 30, 'city': 'NYC'})
    writer.writerow({'name': 'Jane', 'age': 25, 'city': 'LA'})

# Using Pandas
import pandas as pd

# Read CSV
df = pd.read_csv('data.csv')
print(df.head())

# Write CSV
df.to_csv('output.csv', index=False)

# Filter and export
df[df['age'] > 25].to_csv('filtered.csv', index=False)

Java

import com.opencsv.CSVReader;
import com.opencsv.CSVWriter;

// Read CSV
try (CSVReader reader = new CSVReader(new FileReader("data.csv"))) {
    String[] line;
    while ((line = reader.readNext()) != null) {
        System.out.println(Arrays.toString(line));
    }
}

// Write CSV
try (CSVWriter writer = new CSVWriter(new FileWriter("output.csv"))) {
    String[] header = {"Name", "Age", "City"};
    String[] data1 = {"John", "30", "NYC"};
    String[] data2 = {"Jane", "25", "LA"};
    
    writer.writeNext(header);
    writer.writeNext(data1);
    writer.writeNext(data2);
}

Database Operations

# MySQL - Import CSV
LOAD DATA INFILE '/path/to/data.csv'
INTO TABLE users
FIELDS TERMINATED BY ','
ENCLOSED BY '"'
LINES TERMINATED BY '\n'
IGNORE 1 ROWS;

# MySQL - Export CSV
SELECT * FROM users
INTO OUTFILE '/path/to/output.csv'
FIELDS TERMINATED BY ','
ENCLOSED BY '"'
LINES TERMINATED BY '\n';

# PostgreSQL - Import
COPY users FROM '/path/to/data.csv' 
DELIMITER ',' CSV HEADER;

# PostgreSQL - Export
COPY users TO '/path/to/output.csv' 
DELIMITER ',' CSV HEADER;

Data Analysis Example

# Python with Pandas
import pandas as pd
import matplotlib.pyplot as plt

# Load CSV
df = pd.read_csv('sales.csv')

# Basic analysis
print(df.describe())
print(df.groupby('category')['sales'].sum())

# Visualization
df.groupby('month')['sales'].sum().plot(kind='bar')
plt.show()

# Export results
summary = df.groupby('category').agg({
    'sales': ['sum', 'mean', 'count']
})
summary.to_csv('sales_summary.csv')

Best Practices

  • ✅ Always include header row
  • ✅ Quote fields containing commas, quotes, or newlines
  • ✅ Use UTF-8 encoding
  • ✅ Be consistent with delimiters
  • ✅ Handle empty values properly
  • ❌ Avoid complex nested structures
  • ❌ Don't use formulas (Excel-specific)

Common Issues

  • Encoding: Use UTF-8 to avoid character issues
  • Line Endings: Windows (\r\n) vs Unix (\n)
  • Quotes: Escape internal quotes by doubling ""
  • Large Files: Stream processing for big datasets
