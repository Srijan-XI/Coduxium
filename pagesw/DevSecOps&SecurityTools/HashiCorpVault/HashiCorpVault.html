🔐 HashiCorp Vault

Centralized secrets management platform for securing, storing, and controlling access to sensitive data.

Overview

SECRETS SECURITY ENCRYPTION DEVSECOPS

HashiCorp Vault is a tool for securely accessing secrets. It provides a unified interface to any secret, while providing tight access control and recording a detailed audit log. Vault handles API keys, passwords, certificates, encryption keys, and more.

Key Features

  • Secret Storage — Encrypted storage for arbitrary key-value secrets
  • Dynamic Secrets — Generate secrets on-demand for services
  • Data Encryption — Encrypt/decrypt data without storing it
  • Leasing & Renewal — Automatic secret rotation and revocation
  • Revocation — Built-in support for secret revocation
  • Access Control — Fine-grained policies with ACLs
  • Audit Logging — Detailed logs of all Vault operations
  • Multiple Auth Methods — LDAP, Kubernetes, JWT, GitHub, and more

Installation

# macOS
brew tap hashicorp/tap
brew install hashicorp/tap/vault

# Linux (Ubuntu/Debian)
wget -O- https://apt.releases.hashicorp.com/gpg | sudo gpg --dearmor -o /usr/share/keyrings/hashicorp-archive-keyring.gpg
echo "deb [signed-by=/usr/share/keyrings/hashicorp-archive-keyring.gpg] https://apt.releases.hashicorp.com $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/hashicorp.list
sudo apt update && sudo apt install vault

# Docker
docker pull hashicorp/vault

# Verify installation
vault --version

Starting Vault (Dev Mode)

# Start Vault in dev mode (NOT for production)
vault server -dev

# Output shows:
# Root Token: hvs.xxxxxxxxxxxxx
# Unseal Key: (not needed in dev mode)
# Vault Address: http://127.0.0.1:8200

# In another terminal, set environment variables
export VAULT_ADDR='http://127.0.0.1:8200'
export VAULT_TOKEN='hvs.xxxxxxxxxxxxx'

# Check server status
vault status

Basic Secret Operations

# Enable KV secrets engine (v2)
vault secrets enable -path=secret kv-v2

# Write a secret
vault kv put secret/myapp/config \
    username="admin" \
    password="super-secret"

# Read a secret
vault kv get secret/myapp/config

# Read specific field
vault kv get -field=password secret/myapp/config

# List secrets
vault kv list secret/myapp

# Delete secret
vault kv delete secret/myapp/config

# Metadata operations
vault kv metadata get secret/myapp/config
vault kv metadata delete secret/myapp/config

Dynamic Secrets - Database

# Enable database secrets engine
vault secrets enable database

# Configure PostgreSQL connection
vault write database/config/postgresql \
    plugin_name=postgresql-database-plugin \
    allowed_roles="readonly" \
    connection_url="postgresql://{{username}}:{{password}}@localhost:5432/mydb?sslmode=disable" \
    username="vault" \
    password="vault-password"

# Create a role
vault write database/roles/readonly \
    db_name=postgresql \
    creation_statements="CREATE ROLE \"{{name}}\" WITH LOGIN PASSWORD '{{password}}' VALID UNTIL '{{expiration}}'; \
        GRANT SELECT ON ALL TABLES IN SCHEMA public TO \"{{name}}\";" \
    default_ttl="1h" \
    max_ttl="24h"

# Generate credentials
vault read database/creds/readonly

# Output:
# Key            Value
# lease_id       database/creds/readonly/xxxxx
# username       v-root-readonly-xxxxx
# password       A1a-xxxxxxxxxxxxx

Encryption as a Service

# Enable transit secrets engine
vault secrets enable transit

# Create encryption key
vault write -f transit/keys/my-key

# Encrypt data
vault write transit/encrypt/my-key plaintext=$(echo "secret data" | base64)
# Returns: ciphertext:v1:xxxxxxxxxxxxx

# Decrypt data
vault write transit/decrypt/my-key ciphertext="v1:xxxxxxxxxxxxx"
# Returns base64 encoded plaintext

# Rotate encryption key
vault write -f transit/keys/my-key/rotate

# Rewrap data with new key version
vault write transit/rewrap/my-key ciphertext="v1:xxxxxxxxxxxxx"

Access Policies

# Create policy file: app-policy.hcl
path "secret/data/myapp/*" {
  capabilities = ["create", "read", "update", "delete", "list"]
}

path "database/creds/readonly" {
  capabilities = ["read"]
}

path "transit/encrypt/my-key" {
  capabilities = ["update"]
}

path "transit/decrypt/my-key" {
  capabilities = ["update"]
}

# Write policy to Vault
vault policy write app-policy app-policy.hcl

# List policies
vault policy list

# Read policy
vault policy read app-policy

# Create token with policy
vault token create -policy=app-policy

Authentication Methods

# Enable AppRole authentication
vault auth enable approle

# Create AppRole
vault write auth/approle/role/my-app \
    secret_id_ttl=24h \
    token_num_uses=10 \
    token_ttl=20m \
    token_max_ttl=30m \
    secret_id_num_uses=40 \
    policies="app-policy"

# Get Role ID
vault read auth/approle/role/my-app/role-id

# Generate Secret ID
vault write -f auth/approle/role/my-app/secret-id

# Login with AppRole
vault write auth/approle/login \
    role_id="xxxxx" \
    secret_id="xxxxx"

# Kubernetes authentication
vault auth enable kubernetes

vault write auth/kubernetes/config \
    kubernetes_host="https://kubernetes.default.svc:443" \
    kubernetes_ca_cert=@ca.crt \
    token_reviewer_jwt="reviewer-service-account-token"

vault write auth/kubernetes/role/myapp \
    bound_service_account_names=myapp \
    bound_service_account_namespaces=default \
    policies=app-policy \
    ttl=24h

Vault in Docker Compose

version: '3.8'

services:
  vault:
    image: hashicorp/vault:latest
    container_name: vault
    ports:
      - "8200:8200"
    environment:
      VAULT_DEV_ROOT_TOKEN_ID: myroot
      VAULT_DEV_LISTEN_ADDRESS: 0.0.0.0:8200
    cap_add:
      - IPC_LOCK
    volumes:
      - ./vault/data:/vault/data
      - ./vault/logs:/vault/logs
      - ./vault/config:/vault/config
    command: server -dev

  app:
    image: myapp:latest
    environment:
      VAULT_ADDR: http://vault:8200
      VAULT_TOKEN: myroot
    depends_on:
      - vault

Vault Agent for Auto-Auth

# vault-agent.hcl
vault {
  address = "http://127.0.0.1:8200"
}

auto_auth {
  method {
    type = "approle"
    config = {
      role_id_file_path = "/etc/vault/role-id"
      secret_id_file_path = "/etc/vault/secret-id"
    }
  }

  sink {
    type = "file"
    config = {
      path = "/tmp/vault-token"
    }
  }
}

template {
  source      = "/etc/vault/config.tpl"
  destination = "/etc/myapp/config.json"
}

# config.tpl
{
  "database": {
    {{ with secret "database/creds/readonly" }}
    "username": "{{ .Data.username }}",
    "password": "{{ .Data.password }}"
    {{ end }}
  }
}

# Run Vault Agent
vault agent -config=vault-agent.hcl

Python SDK Example

import hvac

# Initialize client
client = hvac.Client(
    url='http://127.0.0.1:8200',
    token='hvs.xxxxxxxxxxxxx'
)

# Check if authenticated
print(client.is_authenticated())

# Write secret
client.secrets.kv.v2.create_or_update_secret(
    path='myapp/config',
    secret=dict(username='admin', password='secret123')
)

# Read secret
secret = client.secrets.kv.v2.read_secret_version(path='myapp/config')
print(secret['data']['data']['password'])

# Dynamic secrets - database
db_creds = client.secrets.database.generate_credentials(name='readonly')
print(f"Username: {db_creds['data']['username']}")
print(f"Password: {db_creds['data']['password']}")

# Encryption
plaintext = b"secret data"
encrypt_response = client.secrets.transit.encrypt_data(
    name='my-key',
    plaintext=plaintext
)
ciphertext = encrypt_response['data']['ciphertext']

# Decryption
decrypt_response = client.secrets.transit.decrypt_data(
    name='my-key',
    ciphertext=ciphertext
)
decrypted = decrypt_response['data']['plaintext']

Kubernetes Integration

# Vault Injector via Annotations
apiVersion: apps/v1
kind: Deployment
metadata:
  name: myapp
spec:
  template:
    metadata:
      annotations:
        vault.hashicorp.com/agent-inject: "true"
        vault.hashicorp.com/role: "myapp"
        vault.hashicorp.com/agent-inject-secret-config: "secret/data/myapp/config"
        vault.hashicorp.com/agent-inject-template-config: |
          {{- with secret "secret/data/myapp/config" -}}
          export DB_USER="{{ .Data.data.username }}"
          export DB_PASS="{{ .Data.data.password }}"
          {{- end }}
    spec:
      serviceAccountName: myapp
      containers:
      - name: myapp
        image: myapp:latest

Best Practices

  • Never use dev mode in production
  • Always enable audit logging
  • Use least privilege for policies
  • Rotate secrets regularly
  • Enable TLS for all communications
  • Use dynamic secrets where possible
  • Implement proper unsealing strategy
  • Monitor Vault metrics and logs
  • Regular backup of Vault data
  • Use namespaces for multi-tenancy (Enterprise)
